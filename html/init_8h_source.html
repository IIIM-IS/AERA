202  SharedLibrary* load(const char* file_name) {
203  SharedLibrary* library = sharedLibrary_.load(file_name);
204  getUserOperatorFunction_ = (void* (*)(const char*))sharedLibrary_.getFunction("GetUserOperatorFunction");
205  return library;

208  void* getFunction(const char* function_name) override {
209  if (getUserOperatorFunction_) {
210  // Try GetUserOperatorFunction first.
211  void* result = getUserOperatorFunction_(function_name);
212  if (result)
213  return result;
214  }
215 
216  // Fall back to searching for the function in the shared library global name space.
217  return sharedLibrary_.getFunction(function_name);
218  }
219 private:
220  SharedLibrary sharedLibrary_;
221  void* (*getUserOperatorFunction_)(const char* function_name);
222 };
