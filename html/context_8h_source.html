108  static void addReference(r_code::Code *destination, uint16 write_index, r_code::Code *referenced_object) {
109 
110  for (uint16 i = 0; i < destination->references_size(); ++i)
111  if (referenced_object == destination->get_reference(i)) {
112 
113  destination->code(write_index) = Atom::RPointer(i);
114  return;
115  }
116 
117  destination->add_reference(referenced_object);
118  destination->code(write_index) = Atom::RPointer(destination->references_size() - 1);
119  }

121  static void addReference(View *destination, uint16 write_index, r_code::Code *referenced_object) { // view references are set in order (index 0 then 1).
122 
123  uint16 r_ptr_index;
124  if (destination->references_[0]) // first ref already in place.
125  r_ptr_index = 1;
126  else
127  r_ptr_index = 0;
128  destination->references_[r_ptr_index] = referenced_object;
129  destination->code(write_index) = Atom::RPointer(r_ptr_index);
130  }

132  template<class C> void copy_structure(C *destination,
133  uint16 write_index,
134  uint16 &extent_index,
135  bool dereference_cptr,
136  int32 pgm_index) const { // assumes the context is a structure; C: Object or View.
137 
138  if ((*this)[0].getDescriptor() == Atom::OPERATOR && Operator::Get((*this)[0].asOpcode()).is_syn()) { // (\ (expression ...)).
139 
140  IPGMContext c = get_child(1);
141  c.copy_member(destination, write_index, extent_index, dereference_cptr, pgm_index);
142  } else {
143 
144  destination->code(write_index++) = (*this)[0];
145 
146  uint16 atom_count = get_children_count();
147  extent_index = write_index + atom_count;
148 
149  switch ((*this)[0].getDescriptor()) {
150  case Atom::C_PTR: // copy members as is (no dereference).
151  if ((*this)[1].getDescriptor() == Atom::CODE_VL_PTR) {
152 
153  Atom a = code_[(*this)[1].asIndex()];
154  if (a.getDescriptor() == Atom::I_PTR)
155  a = code_[a.asIndex()];
156  switch (a.getDescriptor()) {
157  case Atom::OUT_OBJ_PTR:
158  destination->code(write_index++) = Atom::CodeVLPointer(code_[(*this)[1].asIndex()].asIndex());
159  break;
160  case Atom::IN_OBJ_PTR: // TMP: assumes destination is a mk.rdx.
161  destination->code(write_index++) = Atom::RPointer(1 + a.asInputIndex());
162  break;
163  default:
164  destination->code(write_index++) = (*this)[1];
165  break;
166  }
167  } else
168  destination->code(write_index++) = (*this)[1];
169  for (uint16 i = 2; i <= atom_count; ++i)
170  destination->code(write_index++) = (*this)[i];
171  break;
172  case Atom::TIMESTAMP: // copy members as is (no dereference).
173  for (uint16 i = 1; i <= atom_count; ++i)
174  destination->code(write_index++) = (*this)[i];
175  break;
176  case Atom::DURATION: // copy members as is (no dereference).
177  for (uint16 i = 1; i <= atom_count; ++i)
178  destination->code(write_index++) = (*this)[i];
179  break;
180  default:
181  if (is_cmd_with_cptr()) {
182 
183  for (uint16 i = 1; i <= atom_count; ++i) {
184 
185  IPGMContext c = get_child(i);
186  c.copy_member(destination, write_index++, extent_index, i != 2, pgm_index);
187  }
188  } else { // if a pgm is being copied, indicate the starting index of the pgm so that we can turn on code patching and know if a cptr is referencing code inside the pgm (in that case it will not be dereferenced).
189 
190  int32 _pgm_index;
191  if (pgm_index >= 0)
192  _pgm_index = pgm_index;
193  else if ((*this)[0].getDescriptor() == Atom::OBJECT && ((*this)[0].asOpcode() == Opcodes::Pgm || (*this)[0].asOpcode() == Opcodes::AntiPgm))
194  _pgm_index = index_;
195  else
196  _pgm_index = -1;
197 
198  for (uint16 i = 1; i <= atom_count; ++i) {
199 
200  IPGMContext c = get_child(i);
201  c.copy_member(destination, write_index++, extent_index, !(!dereference_cptr && i == 1), _pgm_index);
202  }
203  }
204  break;
205  }
206  }
207  }

209  template<class C> void copy_member(C *destination,
210  uint16 write_index,
211  uint16 &extent_index,
212  bool dereference_cptr,
213  int32 pgm_index) const {
214 
215  switch ((*this)[0].getDescriptor()) {
216  case Atom::I_PTR:
217  case Atom::VALUE_PTR:
218  case Atom::IPGM_PTR: {
219 
220  if (data_ == REFERENCE && index_ == 0) { // points to an object, not to one of its members.
221 
222  addReference(destination, write_index, object_);
223  break;
224  }
225 
226  if (code_[(*this)[0].asIndex()].getDescriptor() == Atom::C_PTR) {
227 
228  if (!dereference_cptr || (pgm_index > 0 && code_[(*this)[0].asIndex() + 1].asIndex() > pgm_index)) { // the latter case occurs when a cptr references code inside a pgm being copied.
229 
230  IPGMContext c = IPGMContext(object_, view_, code_, (*this)[0].asIndex(), (InputLessPGMOverlay *)overlay_, data_);
231  destination->code(write_index) = Atom::IPointer(extent_index);
232  c.copy_structure(destination, extent_index, extent_index, dereference_cptr, pgm_index);
233  break;
234  }
235  }
236 
237  IPGMContext c = dereference();
238  if (c[0].isStructural()) {
239 
240  destination->code(write_index) = Atom::IPointer(extent_index);
241  if (pgm_index > 0 && index_ > pgm_index && data_ == STEM)
242  patch_code(index_, Atom::OutObjPointer(write_index));
243  c.copy_structure(destination, extent_index, extent_index, dereference_cptr, pgm_index);
244  } else
245  destination->code(write_index) = c[0];
246  break;
247  }
248  case Atom::CODE_VL_PTR:
249  switch (code_[(*this)[0].asIndex()].getDescriptor()) {
250  case Atom::PROD_PTR:
251  addReference(destination, write_index, ((InputLessPGMOverlay *)overlay_)->productions_[code_[(*this)[0].asIndex()].asIndex()]);
252  break;
253  case Atom::I_PTR:
254  if (code_[code_[(*this)[0].asIndex()].asIndex()].getDescriptor() == Atom::IN_OBJ_PTR)
255  addReference(destination, write_index, ((PGMOverlay *)overlay_)->getInputObject(code_[code_[(*this)[0].asIndex()].asIndex()].asInputIndex()));
256  else
257  dereference().copy_member(destination, write_index, extent_index, dereference_cptr, pgm_index);
258  break;
259  case Atom::OUT_OBJ_PTR:
260  destination->code(write_index) = Atom::CodeVLPointer(code_[(*this)[0].asIndex()].asIndex());
261  break;
262  case Atom::IN_OBJ_PTR:
263  case Atom::D_IN_OBJ_PTR: {
264 
265  IPGMContext c = dereference();
266  if (c.index_ == 0)
267  addReference(destination, write_index, c.object_);
268  else if (c[0].isStructural()) {
269 
270  destination->code(write_index++) = Atom::IPointer(extent_index);
271  c.copy_structure(destination, extent_index, extent_index, dereference_cptr, pgm_index);
272  } else
273  c.copy_member(destination, write_index, extent_index, dereference_cptr, pgm_index);
274  break;
275  }default:
276  dereference().copy_member(destination, write_index, extent_index, dereference_cptr, pgm_index);
277  break;
278  }
279  break;
280  case Atom::R_PTR:
281  addReference(destination, write_index, object_->get_reference((*this)[0].asIndex()));
282  break;
283  case Atom::PROD_PTR:
284  addReference(destination, write_index, ((InputLessPGMOverlay *)overlay_)->productions_[(*this)[0].asIndex()]);
285  break;
286  case Atom::IN_OBJ_PTR:
287  addReference(destination, write_index, ((PGMOverlay *)overlay_)->getInputObject((*this)[0].asIndex()));
288  break;
289  case Atom::D_IN_OBJ_PTR: {
290  IPGMContext c = dereference();
291  addReference(destination, write_index, c.object_);
292  break;
293  }case Atom::OPERATOR:
294  case Atom::OBJECT:
295  case Atom::MARKER:
296  case Atom::INSTANTIATED_PROGRAM:
297  case Atom::INSTANTIATED_CPP_PROGRAM:
298  case Atom::INSTANTIATED_INPUT_LESS_PROGRAM:
299  case Atom::INSTANTIATED_ANTI_PROGRAM:
300  case Atom::COMPOSITE_STATE:
301  case Atom::MODEL:
302  case Atom::GROUP:
303  case Atom::SET:
304  case Atom::S_SET:
305  case Atom::TIMESTAMP:
306  case Atom::DURATION:
307  case Atom::STRING:
308  destination->code(write_index) = Atom::IPointer(extent_index);
309  if (pgm_index > 0 && index_ > pgm_index && data_ == STEM)
310  patch_code(index_, Atom::OutObjPointer(write_index));
311  copy_structure(destination, extent_index, extent_index, dereference_cptr, pgm_index);
312  break;
313  case Atom::T_WILDCARD:
314  destination->code(write_index) = (*this)[0];
315  break;
316  default:
317  destination->code(write_index) = (*this)[0];
318  if (pgm_index > 0 && index_ > pgm_index && data_ == STEM)
319  patch_code(index_, Atom::OutObjPointer(write_index));
320  break;
321  }
322  }

397  IPGMContext get_child(uint16 index) const {
398 
399  switch (data_) {
400  case STEM:
401  return IPGMContext(object_, view_, code_, index_ + index, (InputLessPGMOverlay *)overlay_, STEM);
402  case REFERENCE:
403  return IPGMContext(object_, view_, code_, index_ + index, (InputLessPGMOverlay *)overlay_, REFERENCE);
404  case VIEW:
405  return IPGMContext(object_, view_, code_, index_ + index, NULL, VIEW);
406  case MKS: {
407 
408  uint16 i = 0;
409  r_code::list<r_code::Code *>::const_iterator m;
410  object_->acq_markers();
411  for (m = object_->markers_.begin(); i < index - 1; ++i, ++m) {
412 
413  if (m == object_->markers_.end()) { // happens when the list has changed after the call to get_children_count.
414 
415  object_->rel_markers();
416  return IPGMContext();
417  }
418  }
419  object_->rel_markers();
420  return IPGMContext(*m, 0);
421  }case VWS: {
422 
423  uint16 i = 0;
424  std::unordered_set<r_code::_View *, r_code::_View::Hash, r_code::_View::Equal>::const_iterator v;
425  object_->acq_views();
426  for (v = object_->views_.begin(); i < index - 1; ++i, ++v) {
427 
428  if (v == object_->views_.end()) { // happens when the list has changed after the call to get_children_count.
429 
430  object_->rel_views();
431  return IPGMContext();
432  }
433  }
434  object_->rel_views();
435  return IPGMContext(object_, (r_exec::View*)*v, &(*v)->code(0), index_ + index, NULL, VIEW);
436  }case VALUE_ARRAY:
437  return IPGMContext(object_, view_, code_, index_ + index, (InputLessPGMOverlay *)overlay_, VALUE_ARRAY);
438  default: // undefined context.
439  return IPGMContext();
440  }
