292  void trace(uint16 i, std::ostream& out, Atom::TraceContext& context) const {
293  Atom& atom = code(i);

295  if (atom.getDescriptor() == Atom::R_PTR) {
296  if (atom.asIndex() < references_size()) {
297  out << " -> " << get_reference(atom.asIndex())->get_oid();
298 #ifdef WITH_DETAIL_OID

301  }
302  else
303  out << " (unassigned) ";
304  }
305  }
