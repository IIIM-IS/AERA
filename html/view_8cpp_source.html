106 float32 View::MorphValue(float32 value, float32 source_thr, float32 destination_thr) {
107 
108  if (value == 0)
109  return destination_thr;
110 
111  if (source_thr > 0) {
112 
113  if (destination_thr > 0) {
114 
115  float32 r = value * destination_thr / source_thr;
116  if (r > 1) // handles precision errors.
117  r = 1;
118  return r;
119  } else
120  return value;
121  }
122  return destination_thr + value;
123 }

125 float32 View::MorphChange(float32 change, float32 source_thr, float32 destination_thr) { // change is always >0.
126 
127  if (source_thr > 0) {
128 
129  if (destination_thr > 0)
130  return change * destination_thr / source_thr;
131  else
132  return change;
133  }
134  return destination_thr + change;
135 }

137 View::View(View *view, Group *group) : r_code::View(), controller_(NULL) {
138 
139  Group *source = view->get_host();
140  object_ = view->object_;
141  memcpy(code_, view->code_, VIEW_CODE_MAX_SIZE * sizeof(Atom));
142  code_[VIEW_OID].atom_ = GetOID();
143  references_[0] = group; // host.
144  references_[1] = source; // origin.
145 
146  // morph ctrl values; NB: res is not morphed as it is expressed as a multiple of the upr.
147  code(VIEW_SLN) = Atom::Float(MorphValue(view->code(VIEW_SLN).asFloat(), source->get_sln_thr(), group->get_sln_thr()));
148  switch (object_->code(0).getDescriptor()) {
149  case Atom::GROUP:
150  code(GRP_VIEW_VIS) = Atom::Float(MorphValue(view->code(GRP_VIEW_VIS).asFloat(), source->get_vis_thr(), group->get_vis_thr()));
151  break;
152  case Atom::NULL_PROGRAM:
153  case Atom::INSTANTIATED_PROGRAM:
154  case Atom::INSTANTIATED_CPP_PROGRAM:
155  case Atom::INSTANTIATED_INPUT_LESS_PROGRAM:
156  case Atom::INSTANTIATED_ANTI_PROGRAM:
157  case Atom::COMPOSITE_STATE:
158  case Atom::MODEL:
159  code(VIEW_ACT) = Atom::Float(MorphValue(view->code(VIEW_ACT).asFloat(), source->get_act_thr(), group->get_act_thr()));
160  break;
161  }
162 
163  reset();
164 }

202 float32 View::update_res() {
203 
204  float32 new_res = get_res();
205  if (new_res == PLUS_INFINITY)
206  return new_res;
207  if (res_changes_ > 0 && acc_res_ != 0)
208  new_res = get_res() + (float32)acc_res_ / (float32)res_changes_;
209  if (--new_res < 0) // decremented by one on behalf of the group (at upr).
210  new_res = 0;
211  code(VIEW_RES) = r_code::Atom::Float(new_res);
212  acc_res_ = 0;
213  res_changes_ = 0;
214  return get_res();
215 }

217 float32 View::update_sln(float32 low, float32 high) {
218 
219  if (sln_changes_ > 0 && acc_sln_ != 0) {
220 
221  float32 new_sln = get_sln() + acc_sln_ / sln_changes_;
222  if (new_sln < 0)
223  new_sln = 0;
224  else if (new_sln > 1)
225  new_sln = 1;
226  code(VIEW_SLN) = r_code::Atom::Float(new_sln);
227  }
228  acc_sln_ = 0;
229  sln_changes_ = 0;
230 
231  float32 sln = get_sln();
232  if (sln < low)
233  ++periods_at_low_sln_;
234  else {
235 
236  periods_at_low_sln_ = 0;
237  if (sln > high)
238  ++periods_at_high_sln_;
239  else
240  periods_at_high_sln_ = 0;
241  }
242  return sln;
243 }

308 NotificationView::NotificationView(Code *origin, Code *destination, Code *marker) : View() {
309 
310  code(VIEW_OPCODE) = r_code::Atom::SSet(ViewOpcode_, VIEW_ARITY); // Structured Set.

316  code(VIEW_ORG) = r_code::Atom::RPointer(1); // origin.
317  code(VIEW_ARITY + 1) = r_code::Atom::Timestamp(); // ijt will be set at injection time.
318  code(VIEW_ARITY + 2) = 0;
319  code(VIEW_ARITY + 3) = 0;
320  references_[0] = destination;
321  references_[1] = origin;
322 
323  reset_init_sln();
