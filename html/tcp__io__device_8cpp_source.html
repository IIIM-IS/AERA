126  r_code::Code* TcpIoDevice<O, S>::eject(r_code::Code* command)
127  {
128  uint16 function = (command->code(CMD_FUNCTION).atom_ >> 8) & 0x000000FF;
129 
130  uint16 args_set_index = command->code(CMD_ARGS).asIndex();
131  if (function == commands_["ready"]) {
132  if (command->code_size() < 2 || command->code(args_set_index + 1).getDescriptor() != Atom::I_PTR ||
133  command->code(command->code(args_set_index + 1).asIndex()).getDescriptor() != Atom::STRING) {
134  cout << "> WARNING: Cannot get identifier as string" << endl;
135  return NULL;
136  }
137 
138  string identifier = Utils::GetString(&command->code(command->code(args_set_index + 1).asIndex()));
139  for (auto it = entities_.begin(); it != entities_.end(); ++it) {
140  if (it->first.compare(identifier) != 0) {
141  continue;
142  }
143 
144  if (!(command->code_size() >= 3 && command->code(args_set_index + 2).getDescriptor() == Atom::R_PTR &&
145  command->references_size() > command->code(args_set_index + 2).asIndex())) {
146  cout << "> WARNING: Cannot get the object for ready " << identifier << endl;
147  return NULL;
148  }
149 
150  Code* obj = command->get_reference(command->code(args_set_index + 2).asIndex());
151  if (!started_) {
152  it->second = obj;
153  startTimeTickThread();
154  }
155  else if (it->second != obj) {
156  // For now, don't allow tracking multiple objects.
157  return NULL;
158  }
159  return command;
160  }
161  // None of the entities available have the same name as the one ejected by AERA
162  return NULL;
163  }
164  // Not ready command, therefore go through all commands and find the appropriate one
165  for (auto cmd = commands_.begin(); cmd != commands_.end(); ++cmd) {
166  if (cmd->second != function) {
167  continue;
168  }
169  Code* obj = command->get_reference(command->code(args_set_index + 1).asIndex());
170  for (auto it = entities_.begin(); it != entities_.end(); ++it) {
171  if (it->second != obj) {
172  continue;
173  }
174  std::unique_ptr<TCPMessage> msg = std::move(constructMessageFromCommand(cmd->first, it->first, command));
175  if (!msg) {
176  cout << "Could not create message from ejected command" << endl;

178  }
179  sendMessage(std::move(msg));
180  break;
181  }

190  {
191  // Get the stored meta data received by the SetupMessage during establishConnection with the correct identifier.
192  std::map<string, MetaData>::iterator stored_meta_data = meta_data_map_.find(cmd_identifier);
193  if (stored_meta_data == meta_data_map_.end()) {
194  cout << "> WARNING: Could not find cmd identifier " << cmd_identifier << " in the MetaData of available commands!" << endl;
195  return NULL;
196  }
197 
198  // Create a new message to send later
199  std::unique_ptr<TCPMessage> msg = std::make_unique<TCPMessage>();
200  msg->set_messagetype(TCPMessage::DATA);
201 
202  DataMessage* data_msg = msg->mutable_datamessage();
203 
204  ProtoVariable* var = data_msg->add_variables();
205 
206  VariableDescription* meta_data = var->mutable_metadata();
207  meta_data->set_datatype(stored_meta_data->second.getType());
208 
209  for (auto it = id_mapping_.begin(); it != id_mapping_.end(); ++it) {
210  if (it->second.compare(entity) == 0) {
211  meta_data->set_entityid(it->first);

213  if (it->second.compare(cmd_identifier) == 0) {
214  meta_data->set_id(it->first);
215  }
216  }
217 
218  auto dim = meta_data->mutable_dimensions();
219  for (uint64_t i = 0; i < stored_meta_data->second.getDimensions().size(); i++) {
220  dim->Add(stored_meta_data->second.getDimensions()[i]);
221  }
222 
223  uint16 args_set_index = cmd->code(CMD_ARGS).asIndex();
224 
225  std::cout << "Eject cmd: " << cmd_identifier << ", enitiy: " << entity << ", Value: " << cmd->code(args_set_index + 2).asFloat() << std::endl;
226 
227 
228  // @todo: Only a single value implemented. For multiple values (dim != [1]) this needs to be extended.
229  switch (stored_meta_data->second.getType()) {
230  case VariableDescription_DataType_BOOL:
231  {
232  char d = (char)cmd->code(args_set_index + 2).asBoolean();
233  var->set_data(std::string(&d));
234  break;
235  }
236  case VariableDescription_DataType_INT64:
237  {
238  // No direct representation as integer, only float32 in AERA
239  int val = (int)cmd->code(args_set_index + 2).asFloat();
240  char d[8];

252  for (size_t i = 0; i < sizeof(double); ++i) {
253  data += d[i];
254  }
255  var->set_data(data);
256  break;

259  {
260  // TODO: Using the .asIndex() function this can be done
261  cout << "> WARNING: String type not implemented, yet" << endl;
262  }
263  }

401  void TcpIoDevice<O, S>::handleSetupMessage(std::unique_ptr<TCPMessage> setup_msg)
402  {
403  auto setup_message = setup_msg->release_setupmessage();
404  // Initialize all entities and store their communication ids in the id_mapping_ map.
405  for (auto it = setup_message->entities().begin(); it != setup_message->entities().end(); ++it) {
406  id_mapping_[it->second] = it->first;
407  entities_[it->first] = NULL;
408  }
409  // Initialize all commands and store their communication ids in the id_mapping_ map.
410  for (auto it = setup_message->commands().begin(); it != setup_message->commands().end(); ++it) {
411  id_mapping_[it->second] = it->first;
412  commands_[it->first] = 0xFFFF;
413  }
414  // Initialize all properties and store their communication ids in the id_mapping_ map.
415  for (auto it = setup_message->objects().begin(); it != setup_message->objects().end(); ++it) {
416  id_mapping_[it->second] = it->first;
417  objects_[it->first] = NULL;
418  }
419  // Add the ready-command, if not received from the environment simulation.
420  if (commands_.find("ready") == commands_.end()) {
421  commands_["ready"] = 0xFFFF;
422  }
423  // Get all command descriptions (their meta-data) and store them for later access during the eject calls.
424  for (int v = 0; v < setup_message->commanddescriptions_size(); ++v) {
425  auto command_desscription = setup_message->commanddescriptions(v);
426  if (commands_.find(command_desscription.name()) == commands_.end()) {
427  cout << "> WARNING: Variable Description found with different name than any commands. This will never be used!" << endl;
428  continue;
429  }
430  meta_data_map_.insert(std::map<string, MetaData>::value_type(command_desscription.name(), MetaData(&command_desscription.description())));
