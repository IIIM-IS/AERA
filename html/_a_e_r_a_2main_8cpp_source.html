103 r_exec::View *build_view(Timestamp time, Code* rstdin) { // this is application dependent WRT view->sync.
104 
105  r_exec::View *view = new r_exec::View();
106  const uint32 arity = VIEW_ARITY; // reminder: opcode not included in the arity.
107  uint16 write_index = 0;
108  uint16 extent_index = arity + 1;
109 
110  view->code(VIEW_OPCODE) = Atom::SSet(r_exec::View::ViewOpcode_, arity);
111  view->code(VIEW_SYNC) = Atom::Float(_View::SYNC_ONCE); // sync on front.
112  view->code(VIEW_IJT) = Atom::IPointer(extent_index); // iptr to injection time.
113  view->code(VIEW_SLN) = Atom::Float(1.0); // sln.
114  view->code(VIEW_RES) = Atom::Float(1); // res is set to 1 upr of the destination group.
115  view->code(VIEW_HOST) = Atom::RPointer(0); // stdin/stdout is the only reference.
116  view->code(VIEW_ORG) = Atom::Nil(); // org.
117 
118  Utils::SetTimestamp(&view->code(extent_index), time);
119 
120  view->references_[0] = rstdin;
121 
122  return view;
123 }

137

141 void test_injection(r_exec::_Mem *mem, float32 n) {
142 
143  Code* rstdin = mem->get_stdin();
144 
145  // mem->timings_report.clear();
146 
147  // uint64 tt1 = 0;

151

153  // v1.reserve(n);

158  auto t0 = r_exec::Now();
159 
160  for (float32 i = 0; i < n; ++i) {
161  // tt1 = r_exec::Now();
162  Code* object = make_object(mem, rstdin, i);
163 
164  auto now = r_exec::Now();
165  // v1.push_back(now - tt1);
166 
167  // Build a fact.
168  // tt2 = now;
169  Code *fact = new r_exec::Fact(object, now, now, 1, 1);
170  // uint64 tt = r_exec::Now();
171  // v2.push_back(tt - tt2);
172 
173  // Build a default view for the fact.
174  // tt3 = tt;
175  r_exec::View *view = build_view(now, rstdin);
176  // tt = r_exec::Now();
177  // v3.push_back(tt - tt3);
178 
179  // Inject the view.
180  // tt4 = tt;
181  view->set_object(fact);
182  mem->inject(view);
183  // v4.push_back(r_exec::Now() - tt4);
184  }
185 
186  auto t1 = r_exec::Now();
187  auto t2 = t1 - t0;
