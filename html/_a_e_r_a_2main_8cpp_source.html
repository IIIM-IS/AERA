102 r_exec::View *build_view(Timestamp time, Code* rstdin) { // this is application dependent WRT view->sync.
103 
104  r_exec::View *view = new r_exec::View();
105  const uint32 arity = VIEW_ARITY; // reminder: opcode not included in the arity.
106  uint16 write_index = 0;
107  uint16 extent_index = arity + 1;
108 
109  view->code(VIEW_OPCODE) = Atom::SSet(r_exec::View::ViewOpcode_, arity);
110  view->code(VIEW_SYNC) = Atom::Float(_View::SYNC_ONCE); // sync on front.
111  view->code(VIEW_IJT) = Atom::IPointer(extent_index); // iptr to injection time.
112  view->code(VIEW_SLN) = Atom::Float(1.0); // sln.
113  view->code(VIEW_RES) = Atom::Float(1); // res is set to 1 upr of the destination group.
114  view->code(VIEW_HOST) = Atom::RPointer(0); // stdin/stdout is the only reference.
115  view->code(VIEW_ORG) = Atom::Nil(); // org.
116 
117  Utils::SetTimestamp(&view->code(extent_index), time);
118 
119  view->references_[0] = rstdin;
120 
121  return view;
122 }
