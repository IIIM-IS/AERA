133 // Upon invocation of take_input() the overlays older than tsc are killed, assuming stc>0; otherwise, overlays live unitl the ipgm dies.
134 // Controllers are built at loading time and at the view's injection time.
135 // Derived classes must expose a function: void reduce(r_code::_View*input); (called by reduction jobs).
136 class r_exec_dll Controller :
137  public _Object {
138 protected:
139  volatile uint32 invalidated_; // 32 bit alignment.
140  volatile uint32 activated_; // 32 bit alignment.
141 
142  std::chrono::microseconds time_scope_;
143 
144  r_code::_View *view_;
145 
146  CriticalSection reductionCS_;
147 
148  virtual void take_input(r_exec::View* /* input */) {}
149  template<class C> void __take_input(r_exec::View *input) { // utility: to be called by sub-classes.
150 
151  ReductionJob<C> *j = new ReductionJob<C>(input, (C *)this);
152 #ifdef WITH_DETAIL_OID
153  OUTPUT_LINE((TraceLevel)0, " make ReductionJob " << j->get_job_id() <<
154  "(" << j->get_detail_oid() << "): controller(" << get_detail_oid() << ")->reduce(View(fact_" <<
155  input->object_->get_oid() << ")) for " << get_core_object()->get_oid());
156 #endif
157  _Mem::Get()->push_reduction_job(j);
158  }
159 
160  Controller(r_code::_View *view);
161 public:
162  virtual ~Controller();
163 
164  std::chrono::microseconds get_tsc() { return time_scope_; }
165 
166  virtual void invalidate() { invalidated_ = 1; }
167  bool is_invalidated() { return invalidated_ == 1; };
168  void activate(bool a) { activated_ = (a ? 1 : 0); }
169  bool is_activated() const { return activated_ == 1; }
170  bool is_alive() const { return invalidated_ == 0 && activated_ == 1; }
171 
172  virtual r_code::Code *get_core_object() const = 0;
173 
174  r_code::Code *get_object() const { return view_->object_; } // return the reduction object (e.g. ipgm, icpp_pgm, cst, mdl).
175  r_exec::View *get_view() const { return (r_exec::View *)view_; } // return the reduction object's view.
176 
177  void _take_input(r_exec::View *input); // called by the rMem at update time and at injection time.
178 
179  virtual void gain_activation() { activate(true); }
180  virtual void lose_activation() { activate(false); }
181 
182  void set_view(View *view);
183 
184  void debug(View* /* input */) {}
