124 bool Class::get_member_index(Metadata *metadata, std::string &name, uint16 &index, Class *&p) const {
125 
126  for (uint16 i = 0; i < things_to_read.size(); ++i)
127  if (things_to_read[i].name_ == name) {
128 
129  index = (has_offset() ? i + 1 : i); // in expressions the lead r-atom is at 0; in objects, members start at 1
130  if (things_to_read[i].used_as_expression()) // the class is: [::a-class]
131  p = NULL;
132  else
133  p = things_to_read[i].get_class(metadata);
134  return true;
135  }
136  return false;
137 }

157 void Class::write(word32 *storage) {
158 
159  storage[0] = atom_.atom_;
160  r_code::Write(storage + 1, str_opcode);
161  uint32 offset = 1 + r_code::GetSize(str_opcode);
162  storage[offset++] = type;
163  storage[offset++] = use_as;
164  storage[offset++] = things_to_read.size();
165  for (uint32 i = 0; i < things_to_read.size(); ++i) {
166 
167  things_to_read[i].write(storage + offset);
168  offset += things_to_read[i].get_size();
169  }
170 }

172 void Class::read(word32 *storage) {
173 
174  atom_ = storage[0];
175  r_code::Read(storage + 1, str_opcode);
176  uint32 offset = 1 + r_code::GetSize(str_opcode);
177  type = (ReturnType)storage[offset++];
178  use_as = (StructureMember::Iteration)storage[offset++];
179  uint32 member_count = storage[offset++];
180  for (uint32 i = 0; i < member_count; ++i) {
181 
182  StructureMember m;
183  m.read(storage + offset);
184  things_to_read.push_back(m);
185  offset += m.get_size();
186  }
187 }
