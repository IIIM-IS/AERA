97 // pgm layout:
98 //
99 // index content
100 //
101 // PGM_TPL_ARGS >iptr to the tpl args set
102 // PGM_INPUTS >iptr to the pattern set
103 // PGM_GUARDS >iptr to the guard set
104 // PGM_PRODS >iptr to the production set
105 // pgm_code[PGM_TPL_ARGS] >tpl arg set #n0
106 // pgm_code[PGM_TPL_ARGS]+1 >iptr to first tpl pattern
107 // ... >...
108 // pgm_code[PGM_TPL_ARGS]+n0 >iptr to last tpl pattern
109 // pgm_code[pgm_code[PGM_TPL_ARGS]+1] >opcode of the first tpl pattern
110 // ... >...
111 // pgm_code[pgm_code[PGM_TPL_ARGS]+n0] >opcode of the last tpl pattern
112 // pgm_code[PGM_INPUTS] >input pattern set #n1
113 // pgm_code[PGM_INPUTS]+1 >iptr to first input pattern
114 // ... >...
115 // pgm_code[PGM_INPUTS]+n1 >iptr to last input pattern
116 // pgm_code[pgm_code[PGM_INPUTS]+1] >opcode of the first input pattern
117 // ... >...
118 // pgm_code[pgm_code[PGM_INPUTS]+n1] >opcode of the last input pattern
119 // ... >...
120 // pgm_code[PGM_GUARDS] >guard set #n2
121 // pgm_code[PGM_GUARDS]+1 >iptr to first guard
122 // ... >...
123 // pgm_code[PGM_GUARDS]+n2 >iptr to last guard
124 // pgm_code[pgm_code[PGM_GUARDS]+1] >opcode of the first guard
125 // ... >...
126 // pgm_code[pgm_code[PGM_GUARDS]+n2] >opcode of the last guard
127 // ... >...
128 // pgm_code[PGM_PRODS] >production set #n3
129 // pgm_code[PGM_PRODS]+1 >iptr to first production
130 // ... >...
131 // pgm_code[PGM_PRODS]+n3 >iptr to last production
132 // pgm_code[pgm_code[PGM_PRODS]+1] >opcode of the first production
133 // ... >...
134 // pgm_code[pgm_code[PGM_PRODS]+n3] >opcode of the last production

170 void InputLessPGMOverlay::patch_tpl_args() { // no rollback on that part of the code.
171  //get_object()->trace();
172  uint16 tpl_arg_set_index = code_[PGM_TPL_ARGS].asIndex(); // index to the set of all tpl patterns.
173  uint16 arg_count = code_[tpl_arg_set_index].getAtomCount();
174  uint16 ipgm_arg_set_index = get_object()->code(IPGM_ARGS).asIndex(); // index to the set of all ipgm tpl args.
175  for (uint16 i = 1; i <= arg_count; ++i) { // pgm_code[tpl_arg_set_index+i] is an iptr to a pattern.
176 
177  Atom &skel_iptr = code_[code_[tpl_arg_set_index + i].asIndex() + 1];
178  uint16 pgm_code_index = code_[tpl_arg_set_index + i].asIndex();
179 
180  patch_tpl_code(pgm_code_index, get_object()->code(ipgm_arg_set_index + i).asIndex());
181  skel_iptr = Atom::IPGMPointer(ipgm_arg_set_index + i); // patch the pgm code with ptrs to the tpl args' actual location in the ipgm code.
182  }
183  //Atom::Trace(code, get_object()->get_reference(0)->code_size());
184 }

186 void InputLessPGMOverlay::patch_tpl_code(uint16 pgm_code_index, uint16 ipgm_code_index) { // patch recursively : in pgm_code[index] with IPGM_PTRs until ::.
187 
188  uint16 atom_count = code_[pgm_code_index].getAtomCount();
189  for (uint16 j = 1; j <= atom_count; ++j) {
190 
191  switch (code_[pgm_code_index + j].getDescriptor()) {
192  case Atom::WILDCARD:
193  code_[pgm_code_index + j] = Atom::IPGMPointer(ipgm_code_index + j);
194  break;
195  case Atom::T_WILDCARD: // leave as is and stop patching.
196  return;
197  case Atom::I_PTR:
198  patch_tpl_code(code_[pgm_code_index + j].asIndex(), get_object()->code(ipgm_code_index + j).asIndex());
199  break;
200  default: // leave as is.
201  break;
202  }
203  }
204 }

209 bool InputLessPGMOverlay::inject_productions() {
210 
211  auto now = Now();
212 
213  bool in_red = false; // if prods are computed by red, we have to evaluate the expression; otherwise, we have to evaluate the prods in the set one by one to be able to reference new objects in this->productions.
214  IPGMContext prods(get_object()->get_reference(0), get_view(), code_, code_[PGM_PRODS].asIndex(), this);
215  if (prods[0].getDescriptor() != Atom::SET) { // prods[0] is not a set: it is assumed to be an expression lead by red.
216 
217  in_red = true;
218  if (!prods.evaluate()) {
219 
220  rollback();
221  productions_.clear();
222  return false;
223  }
224  prods = prods.dereference();
225  }
226  //prods.trace();
227  uint16 production_count = prods.get_children_count();
228  uint16 cmd_count = 0; // cmds to the executive (excl. mod/set) and external devices.
229  for (uint16 i = 1; i <= production_count; ++i) {
230 
231  IPGMContext cmd = prods.get_child_deref(i);
232  if (!in_red && !cmd.evaluate()) {
233 
234  rollback();
235  productions_.clear();
236  return false;
237  }//cmd.trace();
238  IPGMContext function = cmd.get_child_deref(1);
239 
240  // layout of a command:
241  // 0 >icmd opcode
242  // 1 >function
243  // 2 >iptr to the set of arguments
244  // 3 >set
245  // 4 >first arg
246  // or:
247  // 0 >cmd opcode
248  // 1 >function
249  // 2 >iptr to the set of arguments
250  // 3 >psln_thr
251  // 4 >set
252  // 5 >first arg
253 
254  // identify the production of new objects.
255  IPGMContext args = cmd.get_child_deref(2);
256  if (cmd[0].asOpcode() == Opcodes::ICmd) {
257 
258  if (function[0].asOpcode() == Opcodes::Inject ||
259  function[0].asOpcode() == Opcodes::Eject) { // args:[object view]; create an object if not a reference.
260 
261  Code *object;
262  IPGMContext arg1 = args.get_child(1);
263  uint16 index = arg1.getIndex();
264  arg1 = arg1.dereference();
265  // arg1.trace();
266  if (arg1.is_reference())
267  productions_.push_back(arg1.get_object());
268  else {
269 
270  object = _Mem::Get()->build_object(arg1[0]);
271  // arg1.trace();
272  arg1.copy(object, 0);
273  // arg1.trace();
274  // object->trace();
275  productions_.push_back(_Mem::Get()->check_existence(object));
276  }
277  patch_code(index, Atom::ProductionPointer(productions_.size() - 1));
278 
279  ++cmd_count;
280  } else if (function[0].asOpcode() != Opcodes::Mod &&
281  function[0].asOpcode() != Opcodes::Set &&
282  function[0].asOpcode() != Opcodes::Prb)
283  ++cmd_count;
284  } else {
285  ++cmd_count;
286  if (cmd[0].asOpcode() == Opcodes::Cmd)
287  // We will also add the fact of the ejection to the set of productions.
288  ++cmd_count;
289  }
290  }
291 
292  Code *mk_rdx = NULL;
293  uint16 ntf_grp_count = get_view()->get_host()->get_ntf_grp_count();
294 
295  uint16 write_index;
296  uint16 mk_rdx_prod_index;
297  uint16 extent_index;
298  if (ntf_grp_count && cmd_count && (get_object()->code(IPGM_NFR).asBoolean())) { // the productions are command objects (cmd); only injections/ejections and cmds to external devices are notified.
299 
300  mk_rdx = get_mk_rdx(write_index);
301  mk_rdx_prod_index = write_index;
302  mk_rdx->code(write_index++) = Atom::Set(cmd_count);
303  extent_index = write_index + cmd_count;
304  }
305 
306  // all productions have evaluated correctly; now we can execute the commands one by one.
307  for (uint16 i = 1; i <= production_count; ++i) {
308 
309  IPGMContext cmd = prods.get_child_deref(i);
310  IPGMContext function = cmd.get_child_deref(1);
311 
312  // call device functions.
313  IPGMContext args = cmd.get_child_deref(2);
314  if (cmd[0].asOpcode() == Opcodes::ICmd) { // command to the executive.
315 
316  if (function[0].asOpcode() == Opcodes::Inject) { // args:[object view]; retrieve the object and create a view.
317 
318  IPGMContext arg1 = args.get_child(1);
319  arg1.dereference_once();
320  //arg1.trace();
321  Code *object = args.get_child_deref(1).get_object();
322  //object->trace();
323  IPGMContext _view = args.get_child_deref(2);
324  if (_view[0].getAtomCount() != 0) { // regular view (i.e. not |[]).
325 
326  View *view = new View();
327  _view.copy(view, 0);
328  view->set_object(object);
329 
330  view->references_[1] = get_view()->get_host();
331  view->code(VIEW_ORG) = Atom::RPointer(1);
332 
333  _Mem::Get()->inject(view);
334 
335  if (mk_rdx) {
336 
337  mk_rdx->code(write_index++) = Atom::IPointer(extent_index);
338  prods.get_child_deref(i).copy(mk_rdx, extent_index, extent_index);
339  }
340  } else // this allows building objects with no view (case in point: fact on object: only the fact needs to be injected).
341  --cmd_count;
342  } else if (function[0].asOpcode() == Opcodes::Eject) { // args:[object view destination_node]; view.grp=destination grp (stdin ot stdout); retrieve the object and create a view.
343 
344  Code *object = args.get_child_deref(1).get_object();
345 
346  IPGMContext _view = args.get_child_deref(2);
347  View *view = new View();
348  _view.copy(view, 0);
349  view->set_object(object);
350 
351  IPGMContext node = args.get_child_deref(3);
352 
353  _Mem::Get()->eject(view, node[0].getNodeID());
354 
355  if (mk_rdx) {
356 
357  mk_rdx->code(write_index++) = Atom::IPointer(extent_index);
358  prods.get_child_deref(i).copy(mk_rdx, extent_index, extent_index);
359  }
360  } else if (function[0].asOpcode() == Opcodes::Mod) { // args:[iptr-to-cptr value_].
361 
362  void *object;
363  IPGMContext::ObjectType object_type;
364  int16 member_index;
365  uint32 view_oid;
366  args.get_child(1).getMember(object, view_oid, object_type, member_index); // args.get_child(1) is an iptr.
367 
368  if (object) {
369 
370  float32 value = args.get_child_deref(2)[0].asFloat();
371  switch (object_type) {
372  case IPGMContext::TYPE_VIEW: { // add the target and value to the group's pending operations.
373 
374  Group *g = (Group *)object;
375  g->enter();
376  g->pending_operations_.push_back(new Group::Mod(view_oid, member_index, value));
377  g->leave();
378  break;
379  }case IPGMContext::TYPE_OBJECT:
380  ((Code *)object)->mod(member_index, value); // protected internally.
381  break;
382  case IPGMContext::TYPE_GROUP:
383  ((Group *)object)->enter();
384  ((Group *)object)->mod(member_index, value);
385  ((Group *)object)->leave();
386  break;
387  default:
388  rollback();
389  productions_.clear();
390  return false;
391  }
392  }
393  } else if (function[0].asOpcode() == Opcodes::Set) { // args:[iptr-to-cptr value_].
394 
395  void *object;
396  IPGMContext::ObjectType object_type;
397  int16 member_index;
398  uint32 view_oid;
399  args.get_child(1).getMember(object, view_oid, object_type, member_index); // args.get_child(1) is an iptr.
400 
401  if (object) {
402 
403  float32 value = args.get_child_deref(2)[0].asFloat();
404  switch (object_type) {
405  case IPGMContext::TYPE_VIEW: { // add the target and value to the group's pending operations.
406 
407  Group *g = (Group *)object;
408  g->enter();
409  g->pending_operations_.push_back(new Group::Set(view_oid, member_index, value));
410  g->leave();
411  break;
412  }case IPGMContext::TYPE_OBJECT:
413  ((Code *)object)->set(member_index, value); // protected internally.
414  break;
415  case IPGMContext::TYPE_GROUP:
416  ((Group *)object)->enter();
417  ((Group *)object)->set(member_index, value);
418  ((Group *)object)->leave();
419  break;
420  }
421  }
422  } else if (function[0].asOpcode() == Opcodes::NewClass) { // TODO
423 
424  } else if (function[0].asOpcode() == Opcodes::DelClass) { // TODO
425 
426  } else if (function[0].asOpcode() == Opcodes::LDC) { // TODO
427 
428  } else if (function[0].asOpcode() == Opcodes::Swap) { // TODO
429 
430  } else if (function[0].asOpcode() == Opcodes::Prb) { // args:[probe_level,callback_name,msg,set of objects].
431 
432  float32 probe_lvl = args.get_child_deref(1)[0].asFloat();
433  if (probe_lvl < _Mem::Get()->get_probe_level()) {
434 
435  std::string callback_name = Utils::GetString(&args.get_child_deref(2)[0]);
436 
437  Callbacks::Callback callback = Callbacks::Get(callback_name);
438  if (callback) {
439 
440  std::string msg = Utils::GetString(&args.get_child_deref(3)[0]);
441  IPGMContext _objects = args.get_child_deref(4);
442 
443  uint8 object_count = _objects[0].getAtomCount();
444  Code **objects = NULL;
445  if (object_count) {
446 
447  objects = new Code *[object_count];
448  for (uint8 i = 1; i <= object_count; ++i)
449  objects[i - 1] = _objects.get_child_deref(i).get_object();
450  }
451 
452  callback(duration_cast<microseconds>(now - Utils::GetTimeReference()), false, msg.c_str(), object_count, objects);
453  if (object_count)
454  delete[] objects;
455  }
456  }
457  } else if (function[0].asOpcode() == Opcodes::Stop) { // no args.
458 
459  _Mem::Get()->stop();
460  } else { // unknown function.
461 
462  rollback();
463  productions_.clear();
464  return false;
465  }
466  } else if (cmd[0].asOpcode() == Opcodes::Cmd) { // command to an external device, build a cmd object and send it.
467 
468  P<Code> command = _Mem::Get()->build_object(cmd[0]);
469  cmd.copy((Code*)command, 0);
470 
471  Code* executed_command = _Mem::Get()->eject(command);
472 
473  // Build a fact of the command and inject it in stdin. Give the fact an uncertainty range since we don't know when
474  // it will be executed. Otherwise a fact with zero duration may not overlap a fact, making predictions fail.
475  // We offset the beginning of the uncertainty range at a minimum by 2*GetTimeTolerance() from the frame start (the same as SYNC_HOLD)
476  // so that CTPX::reduce will not fail due to "cause in sync with the premise".
477  auto relative_time = duration_cast<microseconds>(now - Utils::GetTimeReference());
478  auto frame_start = now - (relative_time % _Mem::Get()->get_sampling_period());
479  auto after = max(now, frame_start + 2 * Utils::GetTimeTolerance());
480  auto before = frame_start + _Mem::Get()->get_sampling_period();
481  P<Code> fact;
482  if (executed_command) {
483  // Set fact to the efferent copy of the command and inject it.
484  fact = new Fact(executed_command, after, before, 1, 1);
485  View *view = new View(View::SYNC_ONCE, now, 1, 1, _Mem::Get()->get_stdin(), get_view()->get_host(), fact); // SYNC_ONCE, sln=1, res=1,
486  _Mem::Get()->inject(view);
487  string mk_rdx_info = "";
488 #ifdef WITH_DETAIL_OID
489  if (mk_rdx)
490  // We don't know the mk.rdx OID yet, so use the detail OID.
491  mk_rdx_info = " mk.rdx(" + to_string(mk_rdx->get_detail_oid()) + "):";
492 #endif
493  OUTPUT_LINE(IO_DEVICE_INJ_EJT, Utils::RelativeTime(Now()) << mk_rdx_info << " I/O device eject " << fact->get_oid());
494  }
495  else
496  // The command wasn't executed. Set fact to an anti-fact of the original command and record in the mk_rdx.
497  fact = new AntiFact(command, after, before, 1, 1);
498 
499  if (mk_rdx) {
500 
501  // Add the original command.
502  mk_rdx->code(write_index++) = Atom::IPointer(extent_index);
503  prods.get_child_deref(i).copy(mk_rdx, extent_index, extent_index);
504  // Add the fact of the injected command that we just made.
505  mk_rdx->code(write_index++) = Atom::RPointer(mk_rdx->references_size());
506  mk_rdx->add_reference(fact);
507  }
508  }
509  }
510 
511  if (mk_rdx) {
512 
513  mk_rdx->code(mk_rdx_prod_index) = Atom::Set(cmd_count);
514  for (uint16 i = 1; i <= ntf_grp_count; ++i) {
515 
516  NotificationView *v = new NotificationView(get_view()->get_host(), get_view()->get_host()->get_ntf_grp(i), mk_rdx);
517  _Mem::Get()->inject_notification(v, true);
518  }
519 
520  OUTPUT_LINE((TraceLevel)0, Utils::RelativeTime(Now()) << " pgm " << controller_->get_object()->get_oid() <<
521  " -> mk.rdx " << mk_rdx->get_oid());
522  }
523 
524  return true;
525 }

527 Code *InputLessPGMOverlay::get_mk_rdx(uint16 &extent_index) const {
528 
529  uint16 write_index = 0;
530  extent_index = MK_RDX_ARITY + 1;
531 
532  Code *mk_rdx = new r_exec::LObject(_Mem::Get());
533 
534  mk_rdx->code(write_index++) = Atom::Marker(Opcodes::MkRdx, MK_RDX_ARITY);
535  mk_rdx->code(write_index++) = Atom::RPointer(0); // code.
536  mk_rdx->add_reference(get_object());
537  mk_rdx->code(write_index++) = Atom::IPointer(extent_index); // inputs.
538  mk_rdx->code(extent_index++) = Atom::Set(0);
539  mk_rdx->code(write_index++) = Atom::IPointer(extent_index); // productions.

620 void PGMOverlay::patch_input_code(uint16 pgm_code_index, uint16 input_index, uint16 input_code_index, int16 parent_index) { // patch recursively : in pgm_code[pgm_code_index] with (D_)IN_OBJ_PTRs until ::.
621 
622  uint16 atom_count = code_[pgm_code_index].getAtomCount();
623 
624  // Replace the head of a structure by a ptr to the input object.
625  Atom head;
626  if (parent_index < 0)
627  head = code_[pgm_code_index] = Atom::InObjPointer(input_index, input_code_index);
628  else
629  head = code_[pgm_code_index] = Atom::DInObjPointer(parent_index, input_code_index);
630  patch_indices_.push_back(pgm_code_index);
631 
632  // Proceed with the structure's members.
633  for (uint16 j = 1; j <= atom_count; ++j) {
634 
635  uint16 patch_index = pgm_code_index + j;
636  switch (code_[patch_index].getDescriptor()) {
637  case Atom::T_WILDCARD: // leave as is and stop patching.
638  return;
639  case Atom::WILDCARD:
640  if (parent_index < 0)
641  code_[patch_index] = Atom::InObjPointer(input_index, input_code_index + j);
642  else
643  code_[patch_index] = Atom::DInObjPointer(parent_index, input_code_index + j);
644  patch_indices_.push_back(patch_index);
645  break;
646  case Atom::I_PTR: { // sub-structure: go one level deeper in the pattern.
647  uint16 indirection = code_[patch_index].asIndex(); // save the indirection before patching.
648 
649  if (parent_index < 0)
650  code_[patch_index] = Atom::InObjPointer(input_index, input_code_index + j);
651  else
652  code_[patch_index] = Atom::DInObjPointer(parent_index, input_code_index + j);
653  patch_indices_.push_back(patch_index);
654  switch (dereference_in_ptr(head)->code(j).getDescriptor()) { // caution: the pattern points to sub-structures using iptrs. However, the input object may have a rptr instead of an iptr: we have to disambiguate.
655  case Atom::I_PTR: // dereference and recurse.
656  patch_input_code(indirection, input_index, dereference_in_ptr(head)->code(j).asIndex(), parent_index);
657  break;
658  case Atom::R_PTR: // do not dereference and recurse.
659  patch_input_code(indirection, input_index, 0, patch_index);
660  break;
661  default: // shall never happen.
662  break;
663  }
664  break;
665  }default: // leave as is.
666  break;
667  }
668  }
