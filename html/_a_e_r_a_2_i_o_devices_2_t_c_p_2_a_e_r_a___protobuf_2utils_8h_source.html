158  void setMetaData(int entity_id, int id, VariableDescription_DataType t, std::vector<uint64_t> d, std::string opcode_handle = "") {
159  entity_id_ = entity_id;
160  id_ = id;
161  type_ = t;
162  dimensions_ = d;
163  opcode_handle_ = opcode_handle;
164  switch (t)
165  {
166  case 0:
167  type_size_ = 8;
168  break;
169  case 3:
170  type_size_ = 8;
171  break;
172  case 15:
173  type_size_ = 8;
174  break;
175  default:
176  type_size_ = 1;
177  break;
178  }
179  data_size_ = 1;
180  data_length_ = 1;
181  for (int i = 0; i < dimensions_.size(); ++i) {
182  data_size_ *= dimensions_[i] * type_size_;
183  data_length_ *= dimensions_[i];
184  }
185  }
