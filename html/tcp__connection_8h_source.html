112  void enqueue(std::unique_ptr<TCPMessage> t)
113  {
114  std::lock_guard<std::recursive_mutex> lock(mutex_);
115  while (queue_.size() >= max_elements_) {
116  if (queue_.front()->messagetype() == TCPMessage_Type_DATA) {
117  queue_.front()->release_datamessage();
118  }
119  queue_.pop();
120  }
121  queue_.push(std::move(t));
122  }

129  std::unique_ptr<TCPMessage> dequeue()
130  {
131  std::lock_guard<std::recursive_mutex> lock(mutex_);
132  if (queue_.empty()) {
133  return NULL;
134  }
135  //std::cout << name_ << ": Message available, dequeueing" << std::endl;
136  std::cout << queue_.size() << std::endl;
137  std::unique_ptr<TCPMessage> val = std::move(queue_.front());
138  //std::cout << name_ << ": Dequed message of type " << val->messagetype() << std::endl;
139  queue_.pop();
140  return val;
141  }
