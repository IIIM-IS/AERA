125 template<class O, class S> bool TestMem<O, S>::load
126 (std::vector<Code*> *objects, uint32 stdin_oid, uint32 stdout_oid,
127  uint32 self_oid) {
128  // Call the method in the parent class.
129  if (!r_exec::Mem<O, S>::load(objects, stdin_oid, stdout_oid, self_oid))
130  return false;
131 
132  // Find the opcodes we need.
133  ready_opcode_ = r_exec::GetOpcode("ready");
134  set_velocity_y_opcode_ = r_exec::GetOpcode("set_velocity_y");
135  set_force_y_opcode_ = r_exec::GetOpcode("set_force_y");
136  move_y_plus_opcode_ = r_exec::GetOpcode("move_y_plus");
137  move_y_minus_opcode_ = r_exec::GetOpcode("move_y_minus");
138 
139  // Find the objects we need.
140  position_property_ = findObject(objects, "position");
141  position_y_property_ = findObject(objects, "position_y");
142  velocity_y_property_ = findObject(objects, "velocity_y");
143  force_y_property_ = findObject(objects, "force_y");
144  theta_y_property_ = findObject(objects, "theta_y");
145  omega_y_property_ = findObject(objects, "omega_y");
146  primary_group_ = findObject(objects, "primary");
147 
148  // Find the entities we need.
149  for (int i = 0; i <= 9; ++i)
150  yEnt_[i] = findObject(objects, ("y" + to_string(i)).c_str());
151 
152  return true;
153 }

155 template<class O, class S> Code*
156 TestMem<O, S>::findObject(std::vector<Code*> *objects, const char* name) {
157  // Find the object OID.
158  uint32 oid = r_exec::Seed.object_names_.findSymbol(name);
159  if (oid == UNDEFINED_OID)
160  return NULL;
161 
162  // Find the object. (Imitate the code in _Mem::load.)
163  for (uint32 i = 0; i < objects->size(); ++i) {
164  Code *object = (*objects)[i];
165  if (object->get_oid() == oid)
166  return object;
167  }
168 
169  return NULL;
170 }

172 template<class O, class S> Code* TestMem<O, S>::eject(Code *command) {
173  uint16 function = (command->code(CMD_FUNCTION).atom_ >> 8) & 0x000000FF;
174 
175  if (function == ready_opcode_) {
176  uint16 args_set_index = command->code(CMD_ARGS).asIndex();
177  if (command->code_size() >= 2 && command->code(args_set_index + 1).getDescriptor() == Atom::I_PTR &&
178  command->code(command->code(args_set_index + 1).asIndex()).getDescriptor() == Atom::STRING) {
179  string identifier = Utils::GetString(&command->code(command->code(args_set_index + 1).asIndex()));
180 
181  if (identifier == "ball") {
182  if (!(command->code_size() >= 3 && command->code(args_set_index + 2).getDescriptor() == Atom::R_PTR &&
183  command->references_size() > command->code(args_set_index + 2).asIndex())) {
184  cout << "WARNING: Cannot get the object for ready \"ball\"" << endl;
185  return NULL;
186  }
187  if (!velocity_y_property_) {
188  cout << "WARNING: Can't find the velocity_y property" << endl;
189  return NULL;
190  }
191  if (!position_y_property_) {
192  cout << "WARNING: Can't find the position_y property" << endl;
193  return NULL;
194  }
195 
196  Code* obj = command->get_reference(command->code(args_set_index + 2).asIndex());
197  if (!position_y_obj_) {
198  // This is the first call. Remember the object whose position we're reporting.
199  position_y_obj_ = obj;
200  startTimeTickThread();
201  }
202  else {
203  if (position_y_obj_ != obj)
204  // For now, don't allow tracking multiple objects.
205  return NULL;
206  }
207  return command;
208  }
209  if (identifier == "cart-pole") {
210  if (!(command->code_size() >= 3 && command->code(args_set_index + 2).getDescriptor() == Atom::R_PTR &&
211  command->references_size() > command->code(args_set_index + 2).asIndex())) {
212  cout << "WARNING: Cannot get the object for ready \"ball\"" << endl;
213  return NULL;
214  }
215  if (!velocity_y_property_) {
216  cout << "WARNING: Can't find the velocity_y property" << endl;
217  return NULL;
218  }
219  if (!position_y_property_) {
220  cout << "WARNING: Can't find the position_y property" << endl;
221  return NULL;
222  }
223  if (!force_y_property_) {
224  cout << "WARNING: Can't find the force_y property" << endl;
225  return NULL;
226  }
227  if (!omega_y_property_) {
228  cout << "WARNING: Can't find the omega_y property" << endl;
229  return NULL;
230  }
231  if (!theta_y_property_) {
232  cout << "WARNING: Can't find the theta_y property" << endl;
233  return NULL;
234  }
235  Code* obj = command->get_reference(command->code(args_set_index + 2).asIndex());
236  if (!cart_position_y_obj_) {
237  // This is the first call. Remember the object whose position we're reporting.
238  cart_position_y_obj_ = obj;
239  startTimeTickThread();
240  }
241 
242  else {
243  if (cart_position_y_obj_ != obj)
244  // For now, don't allow tracking multiple objects.
245  return NULL;
246  }
247  ofs.open("cart.out");
248  return command;
249  }
250  else {
251  cout << "WARNING: Ignoring unrecognized ready command identifier: " << identifier << endl;
252  return NULL;
253  }
254  }
255  }
256  else if (function == set_velocity_y_opcode_) {
257  if (!velocity_y_property_) {
258  cout << "WARNING: Can't find the velocity_y property" << endl;
259  return NULL;
260  }
261  if (!position_y_property_) {
262  cout << "WARNING: Can't find the position_y property" << endl;
263  return NULL;
264  }
265 
266  auto now = r_exec::Now();
267  lastCommandTime_ = now;
268  uint16 args_set_index = command->code(CMD_ARGS).asIndex();
269  Code* obj = command->get_reference(
270  command->code(args_set_index + 1).asIndex());
271  // Set up position_y_obj_ the same as the ready "ball" command.
272  if (!position_y_obj_) {
273  // This is the first call. Remember the object whose velocity we're setting.
274  position_y_obj_ = obj;
275  startTimeTickThread();
276  }
277  else {
278  if (position_y_obj_ != obj)
279  // For now, don't allow tracking the velocity of multiple objects.
280  return NULL;
281  }
282 
283  velocity_y_ = command->code(args_set_index + 2).asFloat();
284  // Let onTimeTick inject the new velocity_y.
285  return command;
286  }
287  else if (function == set_force_y_opcode_) {
288  if (!velocity_y_property_) {
289  cout << "WARNING: Can't find the velocity_y property" << endl;
290  return NULL;
291  }
292  if (!position_y_property_) {
293  cout << "WARNING: Can't find the position_y property" << endl;
294  return NULL;
295  }
296  if (!omega_y_property_) {
297  cout << "WARNING: Can't find the omega_y property" << endl;
298  return NULL;
299  }
300  if (!theta_y_property_) {
301  cout << "WARNING: Can't find the theta_y property" << endl;
302  return NULL;
303  }
304  if (!force_y_property_) {
305  cout << "WARNING: Can't find the force_y property" << endl;
306  return NULL;
307  }
308  auto now = r_exec::Now();
309  lastCommandTime_ = now;
310  uint16 args_set_index = command->code(CMD_ARGS).asIndex();
311  Code* obj = command->get_reference(
312  command->code(args_set_index + 1).asIndex());
313  // Set up position_y_obj_ the same as the ready "ball" command.
314  if (!cart_position_y_obj_) {
315  // This is the first call. Remember the object whose velocity we're setting.
316  cart_position_y_obj_ = obj;
317  startTimeTickThread();
318  }
319  else {
320  if (cart_position_y_obj_ != obj)
321  // For now, don't allow tracking the velocity of multiple objects.
322  return NULL;
323  }
324 
325  float desired_force = command->code(args_set_index + 2).asFloat();
326  if ((desired_force > max_force) || (desired_force < -max_force)) {
327  if (desired_force > max_force)
328  desired_force = max_force;
329  else if (desired_force < -max_force)
330  desired_force = -max_force;
331 
332  force_y_ = desired_force;
333 
334  uint16 AccCommand;
335  AccCommand = set_force_y_opcode_;
336  // Make (cmd set_force_y [obj: acc:] 1)
337  Code* cmdAcc = new r_exec::LObject(this);
338  cmdAcc->code(0) = Atom::Object(r_exec::GetOpcode("cmd"), 3);
339  cmdAcc->code(1) = Atom::DeviceFunction(AccCommand);
340  cmdAcc->code(2) = Atom::IPointer(4);
341  cmdAcc->code(3) = Atom::Float(1); // psln_thr.
342  cmdAcc->code(4) = Atom::Set(2);
343  cmdAcc->code(5) = Atom::RPointer(0); // obj
344  cmdAcc->code(6) = Atom::Float(force_y_);
345  cmdAcc->set_reference(0, cart_position_y_obj_);
346 
347  return cmdAcc;
348  }
349  else {
350  force_y_ = desired_force;
351  return command;
352  }
353  // Let onTimeTick inject the new force_y.
354  }
355  else if (function == move_y_plus_opcode_ ||
356  function == move_y_minus_opcode_) {
357  if (!position_property_) {
358  cout << "WARNING: Can't find the position property" << endl;
359  return NULL;
360  }
361  for (int i = 0; i <= 9; ++i) {
362  if (!yEnt_[i]) {
363  cout << "WARNING: Can't find the entities y0, y1, etc." << endl;
364  return NULL;
365  }
366  }
367 
368  auto now = r_exec::Now();
369 
370  uint16 args_set_index = command->code(CMD_ARGS).asIndex();
371  Code* obj = command->get_reference(
372  command->code(args_set_index + 1).asIndex());
373  if (!discretePositionObj_) {
374  // This is the first call. Remember the object whose position we're setting.
375  discretePositionObj_ = obj;
376  discretePosition_ = yEnt_[0];
377  startTimeTickThread();
378  }
379  else {
380  if (discretePositionObj_ != obj)
381  // For now, don't allow tracking multiple objects.
382  return NULL;
383  }
384 
385  if (nextDiscretePosition_)
386  // A previous move command is still pending execution.
387  return NULL;
388 
389  // nextDiscretePosition_ will become the position at the next sampling period.
390  lastCommandTime_ = now;
391  const int maxYPosition = 9;
392  if (function == move_y_plus_opcode_) {
393  for (int i = 0; i <= maxYPosition - 1; ++i) {
394  if (discretePosition_ == yEnt_[i])
395  nextDiscretePosition_ = yEnt_[i + 1];
396  }
397  }
398  else if (function == move_y_minus_opcode_) {
399  for (int i = 1; i <= maxYPosition; ++i) {
400  if (discretePosition_ == yEnt_[i])
401  nextDiscretePosition_ = yEnt_[i - 1];
402  }
403  }
404  // Let onTimeTick inject the new position.
405  return command;
406  }
407 
408  return NULL;
409 }

411 template<class O, class S> void TestMem<O, S>::onTimeTick() {
412  auto now = r_exec::Now();
413  if (now <= lastInjectTime_ + get_sampling_period() * 8 / 10)
414  // Not enough time has elapsed to inject a new position.
415  return;
416 
417  if (position_y_obj_) {
418  // We are updating the continuous position_y_.
419  if (lastInjectTime_.time_since_epoch().count() == 0) {
420  // This is the first call, so leave the initial position.
421  }
422  else
423  position_y_ += velocity_y_ * duration_cast<microseconds>(now - lastInjectTime_).count();
424 
425  lastInjectTime_ = now;
426  // Inject the velocity and position.
427  // It seems that velocity_y needs SYNC_HOLD for building models.
428  injectMarkerValueFromIoDevice(
429  position_y_obj_, velocity_y_property_, Atom::Float(velocity_y_),
430  now, now + get_sampling_period(), r_exec::View::SYNC_HOLD);
431  injectMarkerValueFromIoDevice(
432  position_y_obj_, position_y_property_, Atom::Float(position_y_),
433  now, now + get_sampling_period());
434  }
435  if (cart_position_y_obj_) {
436  // We are updating the continuous position_y_.
437  auto now = r_exec::Now();
438  if (now > lastInjectTime_ + get_sampling_period() * 8 / 10) {
439  // Enough time has elapsed to inject a new position.
440  if (lastInjectTime_.time_since_epoch().count() == 0) {
441  // This is the first call, so leave the initial position.
442  }
443  else
444  {
445  auto DeltaK = 1e-6 * duration_cast<microseconds>(now - lastInjectTime_).count();
446  float current_position_y;
447  float current_velocity_y;
448  float current_theta_y;
449  float current_omega_y;
450 
451  current_position_y = next_position_y_;
452  current_velocity_y = next_velocity_y_;
453  current_theta_y = next_theta_y_;
454  current_omega_y = next_omega_y_;
455 
456  //next_velocity_y_ = (((1. / M) * force_y_) * DeltaK) + current_velocity_y;
457  next_velocity_y_ = (0.1 * force_y_) + current_velocity_y;
458  //next_position_y_ = (0.5 * ((1. / M) * force_y_) * DeltaK * DeltaK) + (current_velocity_y * DeltaK) + current_position_y;
459  next_position_y_ = (0.5e-2 * force_y_) + (0.1 * current_velocity_y) + current_position_y;
460  //next_omega_y_ = A*current_theta_y + B*force_y_ + current_omega_y;
461  next_omega_y_ = current_theta_y + 0.1 * force_y_ + current_omega_y;
462  //next_theta_y_ = A * force_y_ + B * current_theta_y + C * current_omega_y;
463  next_theta_y_ = 0.0371943 * force_y_ + 1.05042 * current_theta_y + 0.101675 * current_omega_y;
464  /* Calculate the constants of theta and omega equations
465  #include<iostream>
466  #include<string>
467  #include<algorithm>
468  using namespace std;
469  float max_force = 15; float d = 0; float m = .00001; float M = 1; float b = 1; float L = 100; float g = -10; float DeltaK = 0.1;
470  int main (){ float p = (-g / L);
471  float A_omg = (-g / L) * DeltaK; float B_omg = DeltaK / (M * L);
472  float A_th = ((-2 + exp(sqrt(p) * DeltaK) + exp(sqrt(p) * DeltaK)) / (2 * p));
473  float B_th = ((exp(-sqrt(p) * DeltaK) + exp(sqrt(p) * DeltaK)) / 2);
474  float C_th = ((-sqrt(p) * exp(-sqrt(p) * DeltaK) + sqrt(p) * exp(sqrt(p) * DeltaK)) / (2 * p));
475  cout << "P: " << p << endl; cout << "A_omega: " << A_omg << endl; cout << "B_omega: " << B_omg << endl;
476  cout << "A_theta: " << A_th << endl; cout << "B_theta: " << B_th << endl; cout << "C_theta: " << C_th << endl;
477  return 0;}
478  */
479  }
480  lastInjectTime_ = now;
481 
482 
483  injectMarkerValueFromIoDevice(
484  cart_position_y_obj_, force_y_property_, Atom::Float(force_y_),
485  now, now + get_sampling_period());
486  injectMarkerValueFromIoDevice(
487  cart_position_y_obj_, velocity_y_property_, Atom::Float(next_velocity_y_),
488  now, now + get_sampling_period());
489  injectMarkerValueFromIoDevice(
490  cart_position_y_obj_, position_y_property_, Atom::Float(next_position_y_),
491  now, now + get_sampling_period());
492  injectMarkerValueFromIoDevice(
493  cart_position_y_obj_, theta_y_property_, Atom::Float(next_theta_y_),
494  now, now + get_sampling_period());
495  injectMarkerValueFromIoDevice(
496  cart_position_y_obj_, omega_y_property_, Atom::Float(next_omega_y_),
497  now, now + get_sampling_period());
498 
499  ofs << next_theta_y_ << "," << next_omega_y_ << "," << next_position_y_ << "," << next_velocity_y_ << "," << force_y_ << endl;
500  }
501  }
502  if (discretePositionObj_) {
503  // We are updating the discretePosition_.
504  if (nextDiscretePosition_ &&
505  now >= lastCommandTime_ + get_sampling_period() * 4 / 10) {
506  // Enough time has elapsed from the move command to update the position.
507  discretePosition_ = nextDiscretePosition_;
508  // Clear nextDiscretePosition_ to allow another move command.
509  nextDiscretePosition_ = NULL;
510  }
511 
512  lastInjectTime_ = now;
513  injectMarkerValueFromIoDevice(
514  discretePositionObj_, position_property_, discretePosition_,
515  now, now + get_sampling_period());
516 
517  const microseconds babbleStopTime(3700000);
518  const int maxBabblePosition = 9;
519  if (now - Utils::GetTimeReference() < babbleStopTime) {
520  // Babble.
521  if (discretePosition_ == yEnt_[0])
522  // Reset to the expected value.
523  babbleState_ = 1;
524  else if (discretePosition_ == yEnt_[maxBabblePosition])
525  // Reset to the expected value.
526  babbleState_ = maxBabblePosition + 1;
527  else {
528  ++babbleState_;
529  if (babbleState_ >= maxBabblePosition * 2)
530  babbleState_ = 0;
531  }
532 
533  uint16 nextCommand;
534  if (babbleState_ >= 0 && babbleState_ <= maxBabblePosition - 1)
535  nextCommand = move_y_plus_opcode_;
536  else
537  nextCommand = move_y_minus_opcode_;
538 
539  // Inject (fact (goal (fact (cmd ...))))
540  Code *cmd = new r_exec::LObject(this);
541  cmd->code(0) = Atom::Object(r_exec::GetOpcode("cmd"), 3);
542  cmd->code(1) = Atom::DeviceFunction(nextCommand);
543  cmd->code(2) = Atom::IPointer(4);
544  cmd->code(3) = Atom::Float(1); // psln_thr.
545  cmd->code(4) = Atom::Set(1);
546  cmd->code(5) = Atom::RPointer(0); // obj
547  cmd->set_reference(0, discretePositionObj_);
548 
549  auto after = now + get_sampling_period() + 2 * Utils::GetTimeTolerance();
550  r_exec::Fact* factCmd = new r_exec::Fact(cmd, after, now + 2 * get_sampling_period(), 1, 1);
551  r_exec::Goal* goal = new r_exec::Goal(factCmd, get_self(), NULL, 1);
552  injectFactFromIoDevice(goal, after, now + get_sampling_period(), primary_group_);
553  }
554  }

557 template<class O, class S> void
558 TestMem<O, S>::startTimeTickThread() {
559  if (reduction_core_count_ == 0 && time_core_count_ == 0)
