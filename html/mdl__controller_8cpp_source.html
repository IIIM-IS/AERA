111 bool PrimaryMDLOverlay::reduce(_Fact *input, Fact *f_p_f_imdl, MDLController *req_controller) {
112 
113  OUTPUT_LINE(MDL_IN, Utils::RelativeTime(Now()) << " mdl: " << controller_->get_object()->get_oid() << " <- input: " << input->get_oid());
114  _Fact *input_object;
115  Pred *prediction = input->get_pred();
116  bool is_simulation;
117  if (prediction) {
118 
119  input_object = prediction->get_target();
120  is_simulation = prediction->is_simulation();
121  } else {
122 
123  input_object = input;
124  is_simulation = false;
125  }
126 
127  P<HLPBindingMap> bm = new HLPBindingMap(bindings_);
128  bm->reset_fwd_timings(input_object);
129  switch (bm->match_fwd_lenient(input_object, ((MDLController *)controller_)->get_lhs())) {
130  case MATCH_SUCCESS_POSITIVE: {
131 
132  load_code();
133  P<HLPBindingMap> original_bindings = bindings_;
134  bool is_req = ((MDLController *)controller_)->is_requirement();
135  bool match = false;
136  Fact *f_imdl = ((MDLController *)controller_)->get_f_ihlp(bm, false);
137  RequirementsPair r_p;
138  Fact *ground = NULL;
139  bool wr_enabled = false;
140  ChainingStatus c_s = ((MDLController *)controller_)->retrieve_imdl_fwd(bm, f_imdl, r_p, ground, req_controller, wr_enabled);
141  f_imdl->get_reference(0)->code(I_HLP_WEAK_REQUIREMENT_ENABLED) = Atom::Boolean(wr_enabled);
142  // Use the timestamps in the template parameters that came from the prerequisite model.
143  Timestamp f_imdl_after, f_imdl_before;
144  if (((PrimaryMDLController *)controller_)->get_template_timings(bm, f_imdl_after, f_imdl_before)) {
145  Utils::SetTimestamp<Code>(f_imdl, FACT_AFTER, f_imdl_after);
146  Utils::SetTimestamp<Code>(f_imdl, FACT_BEFORE, f_imdl_before);
147  }
148 
149  bool chaining_allowed = (c_s >= WEAK_REQUIREMENT_ENABLED);
150  bool did_check_simulated_chaining = false;
151  bool check_simulated_chaining_result;
152  vector<BindingResult> bind_results;
153  switch (c_s) {
154  case WEAK_REQUIREMENT_DISABLED:
155  case STRONG_REQUIREMENT_NO_WEAK_REQUIREMENT: // silent monitoring of a prediction that will not be injected.
156  if (is_simulation) { // if there is simulated imdl for the root of one sim in prediction, allow forward chaining.
157 
158  check_simulated_chaining_result = check_simulated_chaining(bm, f_imdl, prediction, bind_results);
159  did_check_simulated_chaining = true;
160  if (check_simulated_chaining_result)
161  chaining_allowed = true;
162  else
163  break;
164  }
165  case NO_REQUIREMENT:
166  if (!chaining_allowed && ((MDLController *)controller_)->has_tpl_args()) // there are tpl args, abort.
167  break;
168  else
169  f_imdl->get_reference(0)->code(I_HLP_WEAK_REQUIREMENT_ENABLED) = Atom::Boolean(false);
170  case STRONG_REQUIREMENT_DISABLED_WEAK_REQUIREMENT: // silent monitoring of a prediction that will not be injected.
171  if (is_simulation) { // if there is simulated imdl for the root of one sim in prediction, allow forward chaining.
172 
173  if (!did_check_simulated_chaining) {
174  check_simulated_chaining_result = check_simulated_chaining(bm, f_imdl, prediction, bind_results);
175  did_check_simulated_chaining = true;
176  }
177  if (check_simulated_chaining_result)
178  chaining_allowed = true;
179  else
180  break;
181  }
182  case WEAK_REQUIREMENT_ENABLED:
183  if (bind_results.size() == 0)
184  // check_simulated_chaining did not create any new bindings, so use the bm that we have been updating.
185  bind_results.push_back(BindingResult(bm, ground));
186  for (size_t i = 0; i < bind_results.size(); ++i) {
187  // evaluate_fwd_guards() uses bindings_, so set it to the binding map.
188  bindings_ = bind_results[i].map_;
189  if (evaluate_fwd_guards()) { // may update bindings_ .
190  f_imdl->set_reference(0, bindings_->bind_pattern(f_imdl->get_reference(0))); // valuate f_imdl from updated binding map.
191  ((PrimaryMDLController *)controller_)->predict(bindings_, input, f_imdl, chaining_allowed, r_p, bind_results[i].ground_);
192  match = true;
193  }
194  }
195  break;
196  }
197  // reset.
198  delete[] code_;
199  code_ = NULL;
200  bindings_ = original_bindings;
201  if (f_p_f_imdl == NULL) // i.e. if reduction not triggered a requirement.
202  store_evidence(input, prediction, is_simulation);
203  return match;
204  }case MATCH_SUCCESS_NEGATIVE: // counter-evidence WRT the lhs.
205  if (f_p_f_imdl == NULL) // i.e. if reduction not triggered a requirement.
206  store_evidence(input, prediction, is_simulation);
207  case MATCH_FAILURE:
208  return false;
209  }
210 }

236 bool SecondaryMDLOverlay::reduce(_Fact *input, Fact *f_p_f_imdl, MDLController *req_controller) { // no caching since no bwd.
237  P<HLPBindingMap> bm = new HLPBindingMap(bindings_);
238  bm->reset_fwd_timings(input);
239  switch (bm->match_fwd_lenient(input, ((MDLController *)controller_)->get_lhs())) {
240  case MATCH_SUCCESS_POSITIVE: {
241 
242  load_code();
243  P<HLPBindingMap> original_bindings = bindings_;
244  bindings_ = bm;
245  bool match = false;
246  Fact *f_imdl = ((MDLController *)controller_)->get_f_ihlp(bm, false);
247  RequirementsPair r_p;
248  Fact *ground = f_p_f_imdl;
249  bool wr_enabled = false;
250  ChainingStatus c_s = ((MDLController *)controller_)->retrieve_imdl_fwd(bm, f_imdl, r_p, ground, req_controller, wr_enabled);
251  f_imdl->get_reference(0)->code(I_HLP_WEAK_REQUIREMENT_ENABLED) = Atom::Boolean(wr_enabled);
252  bool chaining_allowed = (c_s >= NO_REQUIREMENT);
253  switch (c_s) {
254  case WEAK_REQUIREMENT_DISABLED:
255  case STRONG_REQUIREMENT_NO_WEAK_REQUIREMENT: // silent monitoring of a prediction that will not be injected.
256  case NO_REQUIREMENT:
257  if (((MDLController *)controller_)->has_tpl_args()) // there are tpl args, abort.
258  break;
259  else
260  f_imdl->get_reference(0)->code(I_HLP_WEAK_REQUIREMENT_ENABLED) = Atom::Boolean(false);
261  case STRONG_REQUIREMENT_DISABLED_WEAK_REQUIREMENT:
262  case WEAK_REQUIREMENT_ENABLED:
263  if (evaluate_fwd_guards()) { // may update bindings.
264  f_imdl->set_reference(0, bm->bind_pattern(f_imdl->get_reference(0))); // valuate f_imdl from updated bm.
265  ((SecondaryMDLController *)controller_)->predict(bindings_, input, NULL, true, r_p, ground);
266  match = true;
267  }
268  break;
269  }
270  // reset.
271  delete[] code_;
272  code_ = NULL;
273  bindings_ = original_bindings;
274  return match;
275  }case MATCH_SUCCESS_NEGATIVE:
276  case MATCH_FAILURE:
277  return false;
278  }

430 class TemplateTimingsUpdater {
431 public:
439  TemplateTimingsUpdater(_Fact *f_imdl, const _Fact *other_f_imdl, const HLPBindingMap *bm) {
440  f_imdl_ = f_imdl;
441  have_saved_template_timings_ = MDLController::get_imdl_template_timings(
442  f_imdl_->get_reference(0), save_template_after_, save_template_before_,
443  &template_after_ts_index_, &template_before_ts_index_);
444  if (!have_saved_template_timings_)
445  return;
446 
447  // Temporarily make f_imdl_ template timings match the one from _f_imdl so that any difference is ignored.
448  auto other_imdl = other_f_imdl->get_reference(0);
449  auto other_template_set_index = other_imdl->code(I_HLP_TPL_ARGS).asIndex();
450  auto other_template_set_count = other_imdl->code(other_template_set_index).getAtomCount();
451  if (other_template_set_count < 2)
452  return;
453  auto other_template_after_index = other_template_set_index + (other_template_set_count - 1);
454  auto other_template_before_index = other_template_set_index + other_template_set_count;
455 
456  Timestamp other_f_imdl_template_after, other_f_imdl_template_before;
457  if (!getTimestamp(
458  other_imdl, other_template_set_index + (other_template_set_count - 1), other_f_imdl_template_after, bm))
459  return;
460  if (!getTimestamp(other_imdl, other_template_set_index + other_template_set_count,
461  other_f_imdl_template_before, bm))
462  return;
463 
464  // When Match is updated with time interval comparison, it will do this test for strict overlap.
465  if (save_template_after_ < other_f_imdl_template_before && save_template_before_ > other_f_imdl_template_after) {
466  Utils::SetTimestampStruct(f_imdl_->get_reference(0), template_after_ts_index_, other_f_imdl_template_after);
467  Utils::SetTimestampStruct(f_imdl_->get_reference(0), template_before_ts_index_, other_f_imdl_template_before);
468  }
469  }
470 
474  ~TemplateTimingsUpdater() {
475  if (have_saved_template_timings_) {
476  Utils::SetTimestampStruct(f_imdl_->get_reference(0), template_after_ts_index_, save_template_after_);
477  Utils::SetTimestampStruct(f_imdl_->get_reference(0), template_before_ts_index_, save_template_before_);
478  }
479  }

506 ChainingStatus MDLController::retrieve_simulated_imdl_fwd(const HLPBindingMap *bm, Fact *f_imdl, Sim* sim, vector<BindingResult>& results) {
507 
508  uint32 wr_count;
509  uint32 sr_count;
510  uint32 r_count = get_requirement_count(wr_count, sr_count);
511  if (!r_count)
512  return NO_REQUIREMENT;
513  ChainingStatus r;
514  bool save_f_imdl_wr_enabled = f_imdl->get_reference(0)->code(I_HLP_WEAK_REQUIREMENT_ENABLED).asBoolean();
515  if (!sr_count) { // no strong req., some weak req.: true if there is one f->imdl complying with timings and bindings.
516 
517  r = WEAK_REQUIREMENT_DISABLED;
518  requirements_.CS_.enter();
519  auto now = Now();
520  r_code::list<RequirementEntry>::const_iterator e;
521  for (e = simulated_requirements_.positive_evidences_.begin(); e != simulated_requirements_.positive_evidences_.end();) {
522 
523  if ((*e).is_too_old(now)) // garbage collection.
524  e = simulated_requirements_.positive_evidences_.erase(e);
525  else {
526 
527  if ((*e).evidence_->get_pred()->has_simulation(sim)) {
528 
529  _Fact *_f_imdl = (*e).evidence_->get_pred()->get_target();
530  //_f_imdl->get_reference(0)->trace();
531  //f_imdl->get_reference(0)->trace();
532  HLPBindingMap _original(bm); // matching updates the binding map; always start afresh.
533  // Temporarily make f_imdl wr_enabled match the one from _f_imdl so that any difference is ignored.
534  f_imdl->get_reference(0)->code(I_HLP_WEAK_REQUIREMENT_ENABLED) = Atom::Boolean(
535  _f_imdl->get_reference(0)->code(I_HLP_WEAK_REQUIREMENT_ENABLED).asBoolean());
536  TemplateTimingsUpdater timingsUpdater(f_imdl, _f_imdl, &_original);
537  if (_original.match_fwd_strict(_f_imdl, f_imdl)) { // tpl args will be valuated in bm, but not in f_imdl yet.
538 
539  r = WEAK_REQUIREMENT_ENABLED;
540  results.push_back(BindingResult(new HLPBindingMap(_original), (*e).evidence_));
541  // Loop again to check for more matches.
542  }
543  }
544  ++e;
545  }
546  }
547 
548  requirements_.CS_.leave();
549  // Restore.
550  f_imdl->get_reference(0)->code(I_HLP_WEAK_REQUIREMENT_ENABLED) = Atom::Boolean(save_f_imdl_wr_enabled);
551  return r;
