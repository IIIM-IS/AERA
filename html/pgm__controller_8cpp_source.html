231 void AntiPGMController::reduce(r_exec::View *input) {
232 
233  reductionCS_.enter();
234  r_code::list<P<Overlay> >::const_iterator o;
235  for (o = overlays_.begin(); o != overlays_.end();) {
236

238  o = overlays_.erase(o);

247  overlays_.push_back(overlay);
248  successful_match_ = false;

252  if (offspring)

254  }
255  }

259 void AntiPGMController::signal_anti_pgm() {
260 
261  reductionCS_.enter();
262  if (successful_match_) // a signaling job has been spawn in restart(): we are here in an old job during which a positive match occurred: do nothing.
263  successful_match_ = false;
264  else { // no positive match during this job: inject productions and restart.
265 
266  Overlay *overlay = overlays_.front();
267  ((AntiPGMOverlay *)overlay)->inject_productions();
268  overlay->reset(); // reset the first overlay and kill all others.
269  if (!run_once_ && is_alive()) {
270 
271  overlays_.clear();
272  overlays_.push_back(overlay);
273  }
274  }
275  reductionCS_.leave();
276 
277  if (run_once_)
278  invalidate();
279 }
