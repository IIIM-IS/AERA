274  template<class E> void reduce_cache(CriticalSectionList<E> *cache, Fact *f_p_f_imdl, MDLController *controller) {
275 
276  cache->CS_.enter();
277  auto now = Now();
278  r_code::list<E>::const_iterator _e;
279  for (_e = cache->list_.begin(); _e != cache->list_.end();) {
280 
281  if ((*_e).is_too_old(now)) // garbage collection.
282  _e = cache->list_.erase(_e);
283  else if ((*_e).evidence_->get_before() <= now)
284  // Skip the evidence if its time interval ends at exactly now. This is a little
285  // more strict than is_too_old() which only checks get_before() < now.
286  ++_e;
287  else {
288 
289  PrimaryMDLOverlay o(this, bindings_);
290  o.reduce((*_e).evidence_, f_p_f_imdl, controller);
291  ++_e;
292  }
293  }
294  cache->CS_.leave();
295  }
