218  template<class E> void reduce_cache(Cache<E> *cache, Fact *f_p_f_imdl, MDLController *controller) {
219 
220  cache->CS.enter();
221  auto now = Now();
222  r_code::list<E>::const_iterator _e;
223  for (_e = cache->evidences.begin(); _e != cache->evidences.end();) {
224 
225  if ((*_e).is_too_old(now)) // garbage collection.
226  _e = cache->evidences.erase(_e);
227  else if ((*_e).evidence_->get_before() <= now)
228  // Skip the evidence if its time interval ends at exactly now. This is a little
229  // more strict than is_too_old() which only checks get_before() < now.
230  ++_e;
231  else {
232 
233  PrimaryMDLOverlay o(this, bindings_);
234  o.reduce((*_e).evidence_, f_p_f_imdl, controller);
235  ++_e;
236  }
237  }
238  cache->CS.leave();
239  }
