254  template<class E> void reduce_cache(CriticalSectionList<E> *cache, Fact *f_p_f_imdl, MDLController *controller) {
255 
256  cache->CS_.enter();
257  auto now = Now();
258  r_code::list<E>::const_iterator _e;
259  for (_e = cache->list_.begin(); _e != cache->list_.end();) {
260 
261  if ((*_e).is_too_old(now)) // garbage collection.
262  _e = cache->list_.erase(_e);
263  else if ((*_e).evidence_->get_before() <= now)
264  // Skip the evidence if its time interval ends at exactly now. This is a little
265  // more strict than is_too_old() which only checks get_before() < now.
266  ++_e;
267  else {
268 
269  PrimaryMDLOverlay o(this, bindings_);
270  o.reduce((*_e).evidence_, f_p_f_imdl, controller);
271  ++_e;
272  }
273  }
274  cache->CS_.leave();
275  }
