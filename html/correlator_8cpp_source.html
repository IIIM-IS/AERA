239 
240  if (episode_start_ == episode_.size())
241  // no new inputs since last call to get_output => nothing to correlate
242  return new CorrelatorOutput;

244  Episode::iterator it = episode_.begin();
245  if (useEntireHistory)
246  episode_start_ = 0;
247  std::advance(it, episode_start_);
248  winepi.setSeq(it, episode_.end());
249 
250 # define AVG_IN_WINDOW 10
251  int window_size = AVG_IN_WINDOW * (episode_.back().first - episode_[episode_start_].first) / (episode_.size() - episode_start_);
252  winepi.setParams(window_size, 0.1, 0.5, 2); // TODO: find a way to set these appropriately!
253 
254  std::vector<Rule> rules;
255  // clock_t t1 = clock();
256  winepi.algorithm_1(rules); // perform the actual WinEpi algorithm
257 // clock_t t2 = clock();
258 // COUT("Total time taken: " << DIFF_CLOCK(t1,t2) << " seconds.\n");
259 // for(size_t i = 0; i < rules.size(); ++i)
260 // COUT(rules[i].toString());
261 
262  CorrelatorOutput* c = new CorrelatorOutput;
263  c->states_.reserve(rules.size());
264  for (size_t i = 0; i < rules.size(); ++i) {
265  Rule& rule = rules[i];
266  Pattern* p = new Pattern;
267  // LHS and RHS of rules are sorted by OID
268  std::map<int, event_t>::iterator lit = rule.lhs_.G_.begin(), rit = rule.rhs_.G_.begin();
269  while (lit != rule.lhs_.G_.end()) {
270  if (lit->second->get_oid() == rit->second->get_oid()) {
271  p->left_.push_back(lit->second);
272  ++lit;
273  ++rit;
274  }
275  else {
276  p->right_.push_back(rit->second);
277  ++rit;
278  }
279  }
280  for (; rit != rule.rhs_.G_.end(); ++rit) {
281  p->right_.push_back(rit->second);
282  }
283  p->confidence_ = rule.conf_;
284  c->states_.push_back(p);
285  }
286 
287  episode_start_ = episode_.size(); // remember where we left off
288  return c;
289 }

310 // assigns to "result" size-32 vectors representing
311 // all elements in episode, in order, that occur at least twice
312 // runtime: O(N log N) where N = distance(first, last)
313 static void makeNoislessInputs(Episode::const_iterator first, Episode::const_iterator last, LSTMInput& result) {
314 
315  std::set<enc_t> singles(first, last);
316  std::set<enc_t> noise(singles);
317 
318  for (Episode::const_iterator it = first; it != last; ++it) {
319  std::set<enc_t>::iterator sit = singles.find(*it);
320  if (sit != singles.end())
321  singles.erase(sit); // first occurrence
322  else
323  noise.erase(*it); // second occurrence means not noise
324  }
325 
326  result.assign(std::distance(first, last) - noise.size(), LSTMState(32));
327 
328  uint32 i = 0;
329  for (Episode::const_iterator it = first; it != last; ++it) {
330  enc_t code = *it;
331  if (noise.find(code) == noise.end()) {
332  for (uint32 j = 0, b = 1 << 31; j < 32; ++j, b >>= 1)
333  result[i][j] = code & b ? 1 : 0;
334  ++i;
335  }
336  }
337 }

381 // for each set in range [first,last), subtracts all elements from w
382 // stops as soon as no elements of w could be subtracted
383 // returns iterator pointing to first untouched SmallWorld
384 static SmallWorlds::iterator mapDiff(const SmallWorld& w,
385  SmallWorlds::iterator first, SmallWorlds::iterator last) {
386 
387  for (; first != last; ++first) {
388  bool stop = true;
389  for (SmallWorld::const_iterator it = w.begin(); it != w.end(); ++it) {
390  SmallWorld::iterator found = std::find(first->begin(), first->end(), *it);
391  if (found != first->end()) {
392  first->erase(found);
393  stop = false;
394  }
395  }
396  if (stop)
397  break;
398  }
399 
400  return first;
401 }

403 // converts a small worlds representation to a correlator tree
404 static Correlations::iterator swm2corr(
405  SmallWorlds::iterator first, SmallWorlds::iterator last,
406  Correlations::iterator result) {
407 
408  if (first == last) // done
409  return result;
410  if (first->empty()) // boriiing
411  return swm2corr(++first, last, result);
412 
413  // try to see if objects in the first world appear in subsequent worlds
414  SmallWorlds::iterator next = first;
415  SmallWorlds::iterator contextEnd = mapDiff(*first, ++next, last);
416 
417  // no context found; make a pattern
418  if (contextEnd == next) {
419  Pattern* pat = new Pattern();
420  pat->left_.assign(first->begin(), --first->end()); // this is pretty arbitrary
421  pat->right_.assign(--first->end(), first->end());
422  *result++ = pat;
423  }
424  // context found; recurse on it
425  else {
426  IPGMContext* con = new IPGMContext();
427  con->objects_.assign(first->begin(), first->end());
428  con->states_.resize(std::distance(next, contextEnd));
429  Correlations::iterator subEnd = swm2corr(next, contextEnd, con->states_.begin());
430  con->states_.resize(std::distance(con->states_.begin(), subEnd));
431  *result++ = con;
432  }
433 
434  return swm2corr(contextEnd, last, result); // continue after end of context
435 }

437 // calculate the hamming distance between the bit representations of a and b
438 template<class In, class Out>
439 static Out hamdist(In a, In b) {
440 
441  Out dist = 0;
442  for (In x = a ^ b; x; x &= x - 1)
443  ++dist;
444  return dist;
445 }

487 // trains the LSTM network on the episode so far and
488 // returns the found correlations
489 // useEntireHistory determines whether to use entire episode
490 // or only from last call to get_output
491 CorrelatorOutput* Correlator::get_output(bool useEntireHistory) {
492 
493  static bool first_call = true;
494 
495  //dump(); // DEBUG
496 
497  // determine where to start
498  Episode::iterator first = episode_.begin();
499  if (!useEntireHistory)
500  std::advance(first, episode_start_);
501 
502  // remember where we left off
503  episode_start_ = episode_.size();
504 
505  // remove noise and generate LSTM inputs
506  LSTMInput inputs;
507  makeNoislessInputs(first, episode_.end(), inputs);
508 
509  // train the LSTM
510  std::cout << "*** START TRAINING ***" << std::endl; // DEBUG
511  if (first_call) {
512  corcor_.initializeOneStepPrediction(CELLS_PER_BLOCK, NUM_BLOCKS, inputs);
513  corcor_.lstmNetwork_->setRandomWeights(.5);
514  first_call = false;
515  } else
516  corcor_.appendBuffers(inputs);
517  time_t start, current;
518  uint16 epoch;
519  float64 mse;
520  for (time(&start), time(&current), epoch = 0, mse = 0xFFFFFFFF;
521  epoch < NUM_EPOCHS && mse > MSE_THR && difftime(current, start) < TRAIN_TIME_SEC;
522  ++epoch, time(&current)) {
523  mse = corcor_.trainingEpoch(LEARNING_RATE, MOMENTUM);
524  std::cout << epoch << "\t" << mse << std::endl; // DEBUG
525  }
526  std::cout << "#epochs done: " << epoch << std::endl; // DEBUG
527  std::cout << "final MSE: " << mse << std::endl; // DEBUG
528  std::cout << "seconds taken: " << difftime(current, start) << std::endl; // DEBUG
529 
530  // extract rules, and convert to small worlds representation
531  JacobianRules rules;
532  extract_rules(rules, inputs.size());
533  SmallWorlds worlds;
534  rules2worlds(rules, worlds);
535 
536  // the small worlds representation is easily converted to the desired correlator output
537  CorrelatorOutput* c = new CorrelatorOutput();
538  c->states_.resize(worlds.size());
539  Correlations::iterator corrEnd = swm2corr(worlds.begin(), worlds.end(), c->states_.begin());
540  c->states_.resize(std::distance(c->states_.begin(), corrEnd));
541  return c;
542 }

617 void Correlator::extract_rules(JacobianRules& rules, uint32 episode_size) {
618

620  rules.reserve(num_calls);
621  JacobianSlice slice(SLICE_SIZE, LSTMState(32));
622 
623 
624  for (int32 t = num_calls - 1; t >= 0; --t) {
625 
626  // perform sensitivity analysis
627  corcor_.getJacobian(t, t + SLICE_SIZE, slice);
628 
629  // the last one is the target
630  JacobianRule rule;
631  rule.target = findBestMatch(slice.back().begin(), rule.confidence);
632  if (rule.confidence < OBJECT_THR)
633  continue;
634 
635  // the later objects occurred closer to the target, so iterate in reverse
636  JacobianSlice::reverse_iterator it = ++slice.rbegin();
637  JacobianSlice::reverse_iterator end = slice.rend();
638  for (uint16 deltaTime = 1; it != end; ++it, ++deltaTime) {
639

642 
643  // no confidence, no correlation

650  }

654  rule.confidence /= rule.sources.size() + 1;
655  if (rule.confidence >= RULE_THR)
656  rules.push_back(rule);
657  }
658  }
