86 thread_ret thread_function_call TimeCore::Run(void *args) {
87 
88  TimeCore *_this = ((TimeCore *)args);
89 
90  bool run = true;
91  while (run) {
92 
93  P<TimeJob> j = _Mem::Get()->popTimeJob();
94  if (j == NULL)
95  break;
96  if (!j->is_alive()) {
97 
98  j = NULL;
99  continue;
100  }
101 
102  Timestamp target = j->target_time_;
103  Timestamp next_target(microseconds(0));
104  if (target.time_since_epoch().count() == 0) // 0 means ASAP. Control jobs (shutdown) are caught here.
105  run = j->update(next_target);
106  else {
107 
108  auto time_to_wait = duration_cast<microseconds>(target - Now());
109  if (time_to_wait.count() == 0) // right on time: do the job.
110  run = j->update(next_target);
111  else if (time_to_wait.count() > 0) { // early: spawn a delegate to wait for the due time; delegate will die when done.
112 
113  DelegatedCore *d = new DelegatedCore(j->target_time_, time_to_wait, j);
114  d->start(DelegatedCore::Wait);
115  _Mem::Get()->register_time_job_latency(time_to_wait);
116  next_target = Timestamp(seconds(0));
117  } else { // late: do the job and report.
118 
119  run = j->update(next_target);
120  j->report(-time_to_wait);
121  }
122  }
123 
124  while (next_target.time_since_epoch().count() && run) {
125 
126  if (!j->is_alive())
127  break;
128 
129  auto time_to_wait = duration_cast<microseconds>(next_target - Now());
130  next_target = Timestamp(seconds(0));
131  if (time_to_wait.count() == 0) // right on time: do the job.
132  run = j->update(next_target);
133  else if (time_to_wait.count() > 0) { // early: spawn a delegate to wait for the due time; delegate will die when done.
134  // the delegate will handle the next target when it is known (call to update()).
135  DelegatedCore *d = new DelegatedCore(next_target, time_to_wait, j);
136  d->start(DelegatedCore::Wait);
137  } else { // late: do the job and report.
138 
139  run = j->update(next_target);
140  j->report(-time_to_wait);
141  }
142  }
143  j = NULL;
144  }
145 
146  thread_ret_val(0);
147 }

159 thread_ret thread_function_call DelegatedCore::Wait(void *args) {
160 
161  _Mem::Get()->start_core();
162  DelegatedCore *_this = ((DelegatedCore *)args);
163 
164  auto time_to_wait = _this->time_to_wait_;
165  auto target_time = _this->target_time_;
166 
167 wait: _this->timer_.start(time_to_wait);
168  _this->timer_.wait();
169 
170  if (!_this->job_->is_alive())
171  goto end;
172 
173  if (_Mem::Get()->check_state() == _Mem::RUNNING) { // checks for shutdown that could have happened during the wait on timer.
174 
175  while (Now() < target_time); // early, we have to wait; on Windows: timers resolution in ms => poll.
176  target_time = Timestamp(seconds(0));
177  _this->job_->update(target_time);
178  }
179 
180 redo: if (target_time.time_since_epoch().count()) {
181 
182  if (!_this->job_->is_alive())
183  goto end;
184  if (_Mem::Get()->check_state() != _Mem::RUNNING) // checks for shutdown that could have happened during the last update().
185  goto end;
186 
187  time_to_wait = duration_cast<microseconds>(target_time - Now());
188  if (time_to_wait.count() == 0) { // right on time: do the job.
189 
190  _this->job_->update(target_time);
191  goto redo;
192  } else if (time_to_wait.count() < 0) { // late.
193 
194  _this->job_->update(target_time);
195  _this->job_->report(-time_to_wait);
196  goto redo;
197  } else
198  goto wait;
199 }
200 
201  end: _Mem::Get()->shutdown_core();
202  delete _this;
203  thread_ret_val(0);
204 }
