106 bool CalcTimeout(struct timespec &timeout, uint32 ms) {
107 
108  struct timeval now;
109  if (gettimeofday(&now, NULL) != 0)
110  return false;
111 
112  timeout.tv_sec = now.tv_sec + ms / 1000;
113  long us = now.tv_usec + ms % 1000;
114  if (us >= 1000000) {
115  timeout.tv_sec++;
116  us -= 1000000;
117  }
118  timeout.tv_nsec = us * 1000; // usec -> nsec
119  return true;
120 }

171 SharedLibrary *SharedLibrary::load(const char *fileName) {
172 #if defined WINDOWS
173  library_ = LoadLibrary(TEXT(fileName));
174  if (!library_) {
175 
176  DWORD error = GetLastError();
177  std::cerr << "> Error: unable to load shared library " << fileName << " :" << error << std::endl;
178  return NULL;
179  }
180 #elif defined LINUX
181  /*
182  * libraries on Linux are called 'lib<name>.so'
183  * if the passed in fileName does not have those
184  * components add them in.
185  */
186  char *libraryName = (char *)calloc(1, strlen(fileName) + 6 + 1);
187  if (strstr(fileName, "lib") == NULL) {
188  strcat(libraryName, "lib");
189  }
190  strcat(libraryName, fileName);
191  if (strstr(fileName + (strlen(fileName) - 3), ".so") == NULL) {
192  strcat(libraryName, ".so");
193  }
194  library_ = dlopen(libraryName, RTLD_NOW | RTLD_GLOBAL);
195  if (!library_) {
196  std::cout << "> Error: unable to load shared library " << fileName << " :" << dlerror() << std::endl;
197  free(libraryName);
198  return NULL;
199  }
200  free(libraryName);
201 #endif
202  return this;
203 }

352 void Time::Init(uint32 r) {
353 #if defined WINDOWS
354  NTSTATUS nts;
355  HMODULE NTDll = ::LoadLibrary("NTDLL");
356  ULONG actualResolution = 0;
357  if (NTDll) {
358 
359  NSTR pNSTR = (NSTR)::GetProcAddress(NTDll, "NtSetTimerResolution"); // undocumented win xp sys internals
360  if (pNSTR)
361  nts = (*pNSTR)(10 * r, true, &actualResolution); // in 100 ns units
362  }
363  LARGE_INTEGER f;
364  QueryPerformanceFrequency(&f);
365  Period_ = 1000000.0 / f.QuadPart; // in us
366  struct _timeb local_time;
367  _ftime(&local_time);
368  auto now = Timestamp(microseconds((int64)(local_time.time * 1000 + local_time.millitm) * 1000));
369  // The QueryPerformanceCounter in Get() may not start at zero, so subtract it initially.
370  InitTime_ = Timestamp(seconds(0));
371  InitTime_ = now - Get().time_since_epoch();
372 #elif defined LINUX
373  // The steady_clock in Get() may not start at zero, so subtract it initially.
374  InitTime_ = system_clock::now() - duration_cast<microseconds>(steady_clock::now().time_since_epoch());
375 #endif
376 }

389  uint64 t = duration_cast<microseconds>(duration).count();

406 std::string Time::ToString_year(Timestamp timestamp) {
407  uint64 t = duration_cast<microseconds>(timestamp.time_since_epoch()).count();
408 
409  uint64 us = t % 1000;
410  uint64 ms = t / 1000;
411  uint64 s = ms / 1000;
412  ms = ms % 1000;
413 
414  time_t _gmt_time = s;
415  struct tm *_t = gmtime(&_gmt_time);
416 
417  std::string _s = asctime(_t); // _s is: Www Mmm dd hh:mm:ss yyyy but we want: Www Mmm dd yyyy hh:mm:ss:msmsms:ususus
418  std::string year = _s.substr(_s.length() - 5, 4);
419  _s.erase(_s.length() - 6, 5);
420  std::string hh_mm_ss = _s.substr(_s.length() - 9, 8);
421  hh_mm_ss += ":";
422  hh_mm_ss += std::to_string(ms);
423  hh_mm_ss += ":";
424  hh_mm_ss += std::to_string(us);
425 
426  _s.erase(_s.length() - 9, 9);
427  _s += year;
428  _s += " ";
429  _s += hh_mm_ss;
430  _s += " GMT";
431 
432  return _s;
433 }

691 bool Timer::wait(uint32 timeout) {
692 #if defined WINDOWS
693  uint32 r = WaitForSingleObject(t_, timeout);
694  return r == WAIT_TIMEOUT;
695 #elif defined LINUX
696  bool res;
697  struct timespec ttimeout;
698 
699  pthread_mutex_lock(&sematex.mutex);
700  if (timeout == INT_MAX) {
701  res = (pthread_cond_wait(&sematex.semaphore, &sematex.mutex) == 0);
702  }
703  else {
704  CalcTimeout(ttimeout, timeout);
705  res = (pthread_cond_timedwait(&sematex.semaphore, &sematex.mutex, &ttimeout) == 0);
706  }
707  pthread_mutex_unlock(&sematex.mutex);
708  return res;
709 #endif
710 }
